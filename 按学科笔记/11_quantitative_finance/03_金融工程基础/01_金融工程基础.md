## 1. 金融工程的定义

### 1.1 金融工程的定义

**金融工程**（Financial Engineering）是将工程学的方法应用于金融领域，设计、开发和实施金融产品、金融策略和金融系统。

### 1.2 金融工程的特点

1. **数学化**：运用数学方法建模和分析金融问题
2. **工程化**：系统化的方法设计和实施金融产品
3. **创新性**：不断开发新的金融产品和策略
4. **跨学科**：融合数学、统计学、计算机科学和金融学
5. **实用性**：注重实际应用和风险管理

### 1.3 金融工程的发展历程

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy import stats
import cvxpy as cp

class FinancialEngineering:
    """金融工程类"""
    
    def __init__(self):
        """初始化"""
        pass
    
    def calculate_portfolio_return(self, weights, returns):
        """计算投资组合收益率"""
        return np.dot(weights, returns)
    
    def calculate_portfolio_risk(self, weights, cov_matrix):
        """计算投资组合风险"""
        return np.sqrt(np.dot(weights.T, np.dot(cov_matrix, weights)))
    
    def calculate_sharpe_ratio(self, returns, risk_free_rate=0.02):
        """计算夏普比率"""
        excess_returns = returns - risk_free_rate / 252
        return np.mean(excess_returns) / np.std(excess_returns) * np.sqrt(252)
    
    def calculate_max_drawdown(self, returns):
        """计算最大回撤"""
        cumulative = np.cumprod(1 + returns)
        running_max = np.maximum.accumulate(cumulative)
        drawdown = (cumulative - running_max) / running_max
        return np.min(drawdown)
    
    def calculate_var(self, returns, confidence_level=0.05):
        """计算VaR"""
        return np.percentile(returns, confidence_level * 100)
    
    def calculate_cvar(self, returns, confidence_level=0.05):
        """计算CVaR"""
        var = self.calculate_var(returns, confidence_level)
        return np.mean(returns[returns <= var])
    
    def optimize_portfolio(self, expected_returns, cov_matrix, target_return=None):
        """投资组合优化"""
        n = len(expected_returns)
        
        # 定义优化变量
        weights = cp.Variable(n)
        
        # 定义目标函数
        portfolio_variance = cp.quad_form(weights, cov_matrix)
        objective = cp.Minimize(portfolio_variance)
        
        # 定义约束条件
        constraints = [cp.sum(weights) == 1, weights >= 0]
        
        if target_return is not None:
            constraints.append(expected_returns @ weights == target_return)
        
        # 求解优化问题
        problem = cp.Problem(objective, constraints)
        problem.solve()
        
        if problem.status == cp.OPTIMAL:
            return weights.value
        else:
            return None

class FinancialModeling:
    """金融建模类"""
    
    def __init__(self):
        """初始化"""
        self.fe = FinancialEngineering()
    
    def capm_model(self, returns, market_returns, risk_free_rate):
        """CAPM模型"""
        # 计算贝塔系数
        covariance = np.cov(returns, market_returns)[0, 1]
        market_variance = np.var(market_returns)
        beta = covariance / market_variance
        
        # 计算预期收益率
        expected_return = risk_free_rate + beta * (np.mean(market_returns) - risk_free_rate)
        
        return {
            'beta': beta,
            'expected_return': expected_return,
            'alpha': np.mean(returns) - expected_return
        }
    
    def a_risk_model(self, returns, factors):
        """APT风险模型"""
        # 使用多元回归
        X = np.column_stack([np.ones(len(factors)), factors])
        y = returns
        
        # 计算回归系数
        coefficients = np.linalg.lstsq(X, y, rcond=None)[0]
        
        # 计算预期收益率
        expected_return = coefficients[0] + np.dot(coefficients[1:], np.mean(factors, axis=0))
        
        return {
            'factor_loadings': coefficients[1:],
            'expected_return': expected_return,
            'alpha': coefficients[0]
        }
    
    def garch_model(self, returns, p=1, q=1):
        """GARCH模型"""
        n = len(returns)
        omega = 0.01
        alpha = 0.1
        beta = 0.8
        
        # 初始化条件方差
        sigma2 = np.zeros(n)
        sigma2[0] = np.var(returns)
        
        # GARCH过程
        for t in range(1, n):
            sigma2[t] = omega + alpha * returns[t-1]**2 + beta * sigma2[t-1]
        
        return sigma2

class RiskManagement:
    """风险管理类"""
    
    def __init__(self):
        """初始化"""
        self.fe = FinancialEngineering()
    
    def calculate_var_cvar(self, returns, confidence_level=0.05):
        """计算VaR和CVaR"""
        var = self.fe.calculate_var(returns, confidence_level)
        cvar = self.fe.calculate_cvar(returns, confidence_level)
        
        return {
            'var': var,
            'cvar': cvar,
            'confidence_level': confidence_level
        }
    
    def stress_test(self, portfolio_returns, scenarios):
        """压力测试"""
        results = {}
        
        for scenario_name, scenario_returns in scenarios.items():
            # 计算在该情景下的表现
            portfolio_performance = portfolio_returns + scenario_returns
            
            # 计算风险指标
            var = self.fe.calculate_var(portfolio_performance)
            max_dd = self.fe.calculate_max_drawdown(portfolio_performance)
            
            results[scenario_name] = {
                'var': var,
                'max_drawdown': max_dd,
                'mean_return': np.mean(portfolio_performance)
            }
        
        return results
    
    def hedging_strategy(self, spot_price, forward_price, hedge_ratio):
        """对冲策略"""
        # 计算对冲后的收益
        spot_return = (spot_price[1:] - spot_price[:-1]) / spot_price[:-1]
        forward_return = (forward_price[1:] - forward_price[:-1]) / forward_price[:-1]
        
        # 对冲收益
        hedged_return = spot_return + hedge_ratio * forward_return
        
        return hedged_return

class TradingSystem:
    """交易系统类"""
    
    def __init__(self):
        """初始化"""
        self.fe = FinancialEngineering()
        self.rm = RiskManagement()
    
    def generate_signals(self, price_data, window=20):
        """生成交易信号"""
        # 计算移动平均
        short_ma = price_data.rolling(window=window).mean()
        long_ma = price_data.rolling(window=window*2).mean()
        
        # 生成信号
        signals = pd.Series(0, index=price_data.index)
        signals[short_ma > long_ma] = 1  # 买入信号
        signals[short_ma < long_ma] = -1  # 卖出信号
        
        return signals
    
    def backtest(self, price_data, signals):
        """回测交易策略"""
        # 计算收益率
        returns = price_data.pct_change()
        
        # 计算策略收益
        strategy_returns = signals.shift(1) * returns
        
        # 计算累计收益
        cumulative_returns = (1 + strategy_returns).cumprod()
        
        # 计算风险指标
        var = self.fe.calculate_var(strategy_returns)
        max_dd = self.fe.calculate_max_drawdown(strategy_returns)
        
        return {
            'cumulative_returns': cumulative_returns,
            'var': var,
            'max_drawdown': max_dd,
            'total_return': cumulative_returns.iloc[-1] - 1
        }
    
    def execute_trades(self, signals, prices, position_size=1.0):
        """执行交易"""
        trades = []
        position = 0
        
        for i, (signal, price) in enumerate(zip(signals, prices)):
            if signal == 1 and position == 0:
                # 买入
                trades.append({
                    'date': i,
                    'action': 'BUY',
                    'price': price,
                    'position': position + position_size
                })
                position += position_size
            elif signal == -1 and position > 0:
                # 卖出
                trades.append({
                    'date': i,
                    'action': 'SELL',
                    'price': price,
                    'position': position - position_size
                })
                position -= position_size
        
        return trades

    ts = TradingSystem()

def portfolio_optimization_example():
    """投资组合优化示例"""
    # 创建金融工程对象
    fe = FinancialEngineering()

    np.random.seed(42)
    n_assets = 5
    expected_returns = np.random.uniform(0.05, 0.15, n_assets)
    cov_matrix = np.random.uniform(0, 0.05, (n_assets, n_assets))
    cov_matrix = (cov_matrix + cov_matrix.T) / 2
    cov_matrix += np.eye(n_assets) * 0.01
    
    # 无约束优化
    weights = np.ones(n_assets) / n_assets
    min_variance = fe.calculate_portfolio_risk(weights, cov_matrix)
    
    # 最大化夏普比率
    risk_free_rate = 0.02
    excess_returns = expected_returns - risk_free_rate
    sharpe_ratios = excess_returns / np.sqrt(np.diag(cov_matrix))
    max_sharpe_idx = np.argmax(sharpe_ratios)
    
    print("=== 投资组合优化示例 ===")
    print(f"资产数量: {n_assets}")
    print(f"预期收益率: {expected_returns}")
    print(f"无约束最小方差: {min_variance:.4f}")

    random_weights = np.random.dirichlet(np.ones(n_assets), n_portfolios)
    
    portfolio_returns = np.array([fe.calculate_portfolio_return(w, expected_returns) for w in random_weights])
    portfolio_risks = np.array([fe.calculate_portfolio_risk(w, cov_matrix) for w in random_weights])

def risk_management_example():
    """风险管理示例"""
    # 创建风险管理对象
    rm = RiskManagement()

    print(f"99% VaR: {var_99:.4f}")

    scenarios = {
        '熊市': np.random.normal(-0.1, 0.2, 1000),
