```
量化策略框架
```

### 1.2 策略分类

```
量化策略
├── 按盈利逻辑
│   ├── 趋势跟踪 (Trend Following) — 追涨杀跌

├── 按时间频率
│   ├── 高频 (Tick/min) — 毫秒级
│   ├── 日内 (Intraday) — 分钟级
│   ├── 日频 (Daily) — 每日
│   └── 低频 (Weekly/Monthly) — 周/月
│
└── 按市场环境
    ├── 单边做多 — 只能买入
    ├── 多空对冲 — Long/Short
    └── 纯空头 — 只能卖空
```

### 2.2 趋势跟踪信号

**核心逻辑**：认为趋势会延续

```python
def trend_following_signal(price, short_ma=20, long_ma=60):
    """
    均线交叉趋势跟踪策略
    - 短期MA上穿长期MA → 买入信号
    - 短期MA下穿长期MA → 卖出信号
    """
    short_ma_val = price.rolling(short_ma).mean()
    long_ma_val = price.rolling(long_ma).mean()
    
    # 金叉：短期上穿长期
    golden_cross = (short_ma_val > long_ma_val) & (short_ma_val.shift(1) <= long_ma_val.shift(1))
    
    # 死叉：短期下穿长期
    death_cross = (short_ma_val < long_ma_val) & (short_ma_val.shift(1) >= long_ma_val.shift(1))
    
    # 生成信号
    signal = pd.Series(0, index=price.index)
    signal[golden_cross] = 1    # 买入
    signal[death_cross] = -1   # 卖出
    
    return signal


def macd_signal(price, fast=12, slow=26, signal_period=9):
    """
    MACD趋势策略
    """
    ema_fast = price.ewm(span=fast, adjust=False).mean()
    ema_slow = price.ewm(span=slow, adjust=False).mean()
    macd_line = ema_fast - ema_slow
    signal_line = macd_line.ewm(span=signal_period, adjust=False).mean()
    histogram = macd_line - signal_line
    
    # MACD柱由负转正 → 买入
    # MACD柱由正转负 → 卖出
    signal = pd.Series(0, index=price.index)
    signal[histogram > 0] = 1
    signal[histogram < 0] = -1
    
    return signal
```

### 2.3 均值回归信号

**核心逻辑**：认为价格会回归价值

```python
def mean_reversion_signal(price, window=20, num_std=2):
    """
    均值回归策略（布林带）
    - 价格触及下轨 → 买入（超卖）
    - 价格触及上轨 → 卖出（超买）
    """
    ma = price.rolling(window).mean()
    std = price.rolling(window).std()
    
    upper = ma + num_std * std
    lower = ma - num_std * std
    
    signal = pd.Series(0, index=price.index)
    signal[price < lower] = 1   # 超卖，买入
    signal[price > upper] = -1  # 超买，卖出
    
    return signal


def zscore_signal(price, window=20):
    """
    Z-Score均值回归策略
    Z = (价格 - 均值) / 标准差
    """
    ma = price.rolling(window).mean()
    std = price.rolling(window).std()
    zscore = (price - ma) / std
    
    signal = pd.Series(0, index=price.index)
    signal[zscore < -2] = 1    # 严重超卖
    signal[zscore > 2] = -1    # 严重超买
    signal[(zscore > -0.5) & (zscore < 0.5)] = 0  # 回归均值，持平
    
    return signal
```

### 2.4 事件驱动信号

```python
def event_driven_signal(price, earnings_dates, earnings_shock=0.05):
    """
    盈利公告事件策略
    - 盈利超预期 → 买入
    - 盈利不及预期 → 卖出
    """
    signal = pd.Series(0, index=price.index)
    
    for date in earnings_dates:
        if date in price.index:
            # 简化处理：假设已知超预期幅度
            shock = np.random.choice([earnings_shock, -earnings_shock])
            idx = price.index.get_loc(date)
            signal.iloc[idx] = 1 if shock > 0 else -1
    
    return signal
```

## 3. 组合构建

### 3.1 等权重组合

```python
def equal_weight_portfolio(stocks, signal):
    """
    等权重配置
    所有signal=1的股票平均分配资金
    """
    selected = stocks[signal == 1]
    n = len(selected)
    
    if n == 0:
        return pd.Series(0, index=stocks.index)
    
    weights = pd.Series(1/n, index=selected)
    return weights
```

### 3.2 市值加权组合

```python
def market_cap_weighted_portfolio(stocks, market_cap, signal):
    """
    市值加权配置
    信号=1的股票按市值比例分配
    """
    selected = stocks[signal == 1]
    selected_mcap = market_cap[signal == 1]
    
    if len(selected) == 0:
        return pd.Series(0, index=stocks.index)
    
    total_mcap = selected_mcap.sum()
    weights = selected_mcap / total_mcap
    
    return weights
```

### 3.3 因子加权组合

```python
def factor_weighted_portfolio(stocks, factor_values, signal, top_n=10):
    """
    因子值加权配置
    按因子值排序，取Top N，给予权重
    """
    selected = stocks[signal == 1]
    selected_factor = factor_values[signal == 1]
    
    # 取因子值最高的top_n
    top_stocks = selected_factor.nlargest(top_n).index
    
    # 因子值归一化权重
    top_factor = factor_values.loc[top_stocks]
    weights = top_factor / top_factor.sum()
    
    result = pd.Series(0, index=stocks.index)
    result.loc[top_stocks] = weights
    
    return result
```

## 4. 风险管理

### 4.1 仓位管理

```python
def kelly_criterion(win_rate, avg_win, avg_loss):
    """
    凯利公式：最优仓位比例
    f* = (p × b - q) / b
    p: 胜率
    q: 败率 (1-p)
    b: 盈亏比 (avg_win / avg_loss)
    
    警告：实际使用建议用半凯利 (f*/2)
    """
    p = win_rate
    q = 1 - p
    b = avg_win / avg_loss
    
    f_star = (p * b - q) / b
    
    # 限制仓位上限
    f_star = min(max(f_star, 0), 0.3)  # 最大30%仓位
    
    return f_star


def fixed_fractional(position_size, fraction=0.1):
    """
    固定比例仓位
    每次交易使用账户的固定比例
    """
    return fraction


def volatility_targeting(account_value, target_vol, realized_vol):
    """
    波动率目标仓位
    目标：组合波动率维持在目标水平
    """
    if realized_vol == 0:
        return 0.1
    
    # 调整系数
    vol_adj = target_vol / realized_vol
    
    # 限制调整范围
    vol_adj = min(max(vol_adj, 0.2), 3.0)
    
    position = account_value * vol_adj
    return min(position, account_value * 0.3)  # 最大30%
```

### 4.2 止损机制

```python
class StopLossManager:
    def __init__(self, stop_loss_pct=0.05, trailing_stop=False):
        self.stop_loss_pct = stop_loss_pct
        self.trailing_stop = trailing_stop
        self.high_water_mark = {}
    
    def check_stop_loss(self, position, current_price):
        """
        检查是否触发止损
        """
        entry_price = position['entry_price']
        position_value = position['shares'] * current_price
        
        if self.trailing_stop:
            # 追踪止损：记录最高价
            self.high_water_mark[position['symbol']] = max(
                self.high_water_mark.get(position['symbol'], entry_price),
                current_price
            )
            stop_price = self.high_water_mark[position['symbol']] * (1 - self.stop_loss_pct)
        else:
            # 固定止损
            stop_price = entry_price * (1 - self.stop_loss_pct)
        
        should_stop = current_price <= stop_price
        
        return should_stop, stop_price
```

```python
class QuantStrategy:
    def __init__(self, name, initial_capital=100000):
        self.name = name
        self.capital = initial_capital
        self.positions = {}  # symbol -> {shares, entry_price}
        self.portfolio_value = [initial_capital]
        self.trades = []
    
    def generate_signals(self, market_data):
        """生成交易信号"""
        raise NotImplementedError
    
    def select_stocks(self, signals, market_data, top_n=10):
        """选股"""
        raise NotImplementedError
    
    def allocate_weights(self, selected_stocks, market_data):
        """分配权重"""
        raise NotImplementedError
    
    def rebalance(self, weights, market_data):
        """调仓"""
        for symbol, target_weight in weights.items():
            target_value = self.portfolio_value[-1] * target_weight
            current_value = self.positions.get(symbol, {}).get('shares', 0) * market_data.loc[symbol, 'close']
            
            diff_value = target_value - current_value
            
            if abs(diff_value) > 100:  # 最小交易阈值
                shares = int(diff_value / market_data.loc[symbol, 'close'])
                self.execute_trade(symbol, shares, market_data.loc[symbol, 'close'])
    
    def execute_trade(self, symbol, shares, price):
        """执行交易"""
        if shares > 0:
            # 买入
            cost = shares * price * (1 + 0.0003)  # 手续费
            if cost <= self.capital:
                self.positions[symbol] = {'shares': shares, 'entry_price': price}
                self.capital -= cost
                self.trades.append({'symbol': symbol, 'action': 'BUY', 'shares': shares, 'price': price})
        elif shares < 0:
            # 卖出
            symbol_pos = self.positions.get(symbol, {}).get('shares', 0)
            shares_to_sell = min(abs(shares), symbol_pos)
            if shares_to_sell > 0:
                proceeds = shares_to_sell * price * (1 - 0.0003 - 0.001)  # 手续费+印花税
                self.positions[symbol]['shares'] -= shares_to_sell
                self.capital += proceeds
                self.trades.append({'symbol': symbol, 'action': 'SELL', 'shares': shares_to_sell, 'price': price})
    
    def update_portfolio_value(self, market_data):
        """更新组合市值"""
        position_value = sum(
            pos['shares'] * market_data.loc[symbol, 'close']
            for symbol, pos in self.positions.items() if pos['shares'] > 0
        )
        total_value = self.capital + position_value
        self.portfolio_value.append(total_value)
    
    def run(self, market_data, rebalance_freq='20D'):
        """运行策略"""
        for i in range(20, len(market_data)):
            date = market_data.index[i]
            data = market_data.iloc[:i+1]
            
            if i % 20 == 0:  # 每20天调仓
                signals = self.generate_signals(data)
                selected = self.select_stocks(signals, data)
                weights = self.allocate_weights(selected, data)
                self.rebalance(weights, data.iloc[-1])
            
            self.update_portfolio_value(data.iloc[-1])
        
        return self.get_performance()
    
    def get_performance(self):
        """计算绩效"""
        returns = pd.Series(self.portfolio_value).pct_change().dropna()
        return {
            'total_return': (self.portfolio_value[-1] / self.portfolio_value[0]) - 1,
            'annual_return': returns.mean() * 252,
            'annual_vol': returns.std() * np.sqrt(252),
            'sharpe_ratio': (returns.mean() * 252) / (returns.std() * np.sqrt(252)),
            'max_drawdown': self.calculate_max_drawdown(),
            'trade_count': len(self.trades)
        }
    
    def calculate_max_drawdown(self):
        """最大回撤"""
        portfolio = pd.Series(self.portfolio_value)
        running_max = portfolio.expanding().max()
        drawdown = (portfolio - running_max) / running_max
        return drawdown.min()
```
