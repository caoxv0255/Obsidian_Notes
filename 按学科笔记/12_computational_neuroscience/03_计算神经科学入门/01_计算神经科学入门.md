## 2. 计算神经科学的主要方法

### 2.1 神经动力学建模

#### Hodgkin-Huxley模型

**基本原理**：
- 描述动作电位的产生机制
- 考虑多种离子通道
- 建立电学等效电路

**数学表达**：
$$C_m \frac{dV}{dt} = -\sum_{i} g_i(V - E_i) - I_{syn} + I_{ext}$$

其中：
- $C_m$：膜电容
- $V$：膜电位
- $g_i$：离子电导
- $E_i$：离子平衡电位
- $I_{syn}$：突触电流
- $I_{ext}$：外部电流

#### Izhikevich模型

**基本原理**：
- 简化的神经元模型
- 保持Hodgkin-Huxley模型的主要特征
- 计算效率更高

**数学表达**：
$$\begin{cases}
\dot{v} = 0.04v^2 + 5v + 140 - u + I \\
\dot{u} = a(bv - u)
\end{cases}$$

其中：
- $v$：膜电位
- $u$：恢复变量
- $I$：输入电流
- $a, b$：参数

#### Python实现
```python
import numpy as np
import matplotlib.pyplot as plt

class HodgkinHuxleyModel:
    """Hodgkin-Huxley模型"""
    
    def __init__(self):
        """初始化"""
        # 膜电容
        self.C_m = 1.0  # μF/cm²
        
        # 离子电导
        self.g_Na = 120.0  # mS/cm²
        self.g_K = 36.0    # mS/cm²
        self.g_L = 0.3     # mS/cm²
        
        # 离子平衡电位
        self.E_Na = 115.0  # mV
        self.E_K = -12.0   # mV
        self.E_L = 10.6   # mV
        
        # 时间常数
        self.tau_m = 1.0  # ms
        
        # 初始条件
        self.v = -65.0  # mV
        self.n = 0.32  # m
        self.m = 0.05  # m
        self.h = 0.60  # m
    
    def alpha_n(self, v):
        """n的alpha函数"""
        return 0.01 * (v + 55.0) / (1.0 - np.exp(-(v + 55.0) / 10.0))
    
    def beta_n(self, v):
        """n的beta函数"""
        return 0.125 * np.exp(-(v + 65.0) / 80.0)
    
    def alpha_m(self, v):
        """m的alpha函数"""
        return 0.1 * (v + 40.0) / (1.0 - np.exp(-(v + 40.0) / 10.0))
    
    def beta_m(self, v):
        """m的beta函数"""
        return 4.0 * np.exp(-(v + 65.0) / 18.0)
    
    def alpha_h(self, v):
        """h的alpha函数"""
        return 0.07 * np.exp(-(v + 65.0) / 20.0)
    
    def beta_h(self, v):
        """h的beta函数"""
        return 1.0 / (1.0 + np.exp(-(v + 35.0) / 10.0))
    
    def simulate(self, I_ext, dt=0.01, t_max=100):
        """模拟"""
        n_steps = int(t_max / dt)
        v = np.zeros(n_steps)
        n = np.zeros(n_steps)
        m = np.zeros(n_steps)
        h = np.zeros(n_steps)
        
        v[0] = self.v
        n[0] = self.n
        m[0] = self.m
        h[0] = self.h
        
        for i in range(1, n_steps):
            # 更新离子通道
            n[i] = n[i-1] + dt * (self.alpha_n(v[i-1]) * (1 - n[i-1]) - self.beta_n(v[i-1]) * n[i-1])
            m[i] = m[i-1] + dt * (self.alpha_m(v[i-1]) * (1 - m[i-1]) - self.beta_m(v[i-1]) * m[i-1])
            h[i] = h[i-1] + dt * (self.alpha_h(v[i-1]) * (1 - h[i-1]) - self.beta_h(v[i-1]) * h[i-1])
            
            # 计算离子电流
            I_Na = self.g_Na * m[i]**3 * h[i] * (v[i-1] - self.E_Na)
            I_K = self.g_K * n[i]**4 * (v[i-1] - self.E_K)
            I_L = self.g_L * (v[i-1] - self.E_L)
            
            # 更新膜电位
            dv_dt = (I_ext[i] - I_Na - I_K - I_L) / self.C_m
            v[i] = v[i-1] + dt * dv_dt
        
        return v, n, m, h

class IzhikevichModel:
    """Izhikevich模型"""
    
    def __init__(self, a=0.02, b=0.2, c=-65, d=8):
        """初始化"""
        self.a = a
        self.b = b
        self.c = c
        self.d = d
        
        # 初始条件
        self.v = -65.0  # mV
        self.u = -13.0  # mV
    
    def simulate(self, I_ext, dt=0.01, t_max=100):
        """模拟"""
        n_steps = int(t_max / dt)
        v = np.zeros(n_steps)
        u = np.zeros(n_steps)
        
        v[0] = self.v
        u[0] = self.u
        
        for i in range(1, n_steps):
            # 更新膜电位
            v[i] = v[i-1] + dt * (0.04 * v[i-1]**2 + 5 * v[i-1] + 140 - u[i-1] + I_ext[i])
            
            # 更新恢复变量
            u[i] = u[i-1] + dt * (self.a * (self.b * v[i] - u[i]))
            
            # 检测动作电位
            if v[i] >= 30:
                v[i] = self.c
                u[i] = u[i] + self.d
        
        return v, u

# 使用示例
if __name__ == "__main__":

    hh = HodgkinHuxleyModel()
    
    # 创建Izhikevich模型对象

    # 生成外部电流

    t_max = 100
    n_steps = int(t_max / dt)
    t = np.arange(0, t_max, dt)
    I_ext = np.zeros(n_steps)
    I_ext[1000:2000] = 10  # 注入电流

    # 模拟Izhikevich模型
    v_izh, u = izh.simulate(I_ext, dt, t_max)
    
    # 绘制结果

    plt.subplot(2, 2, 1)
    plt.plot(t, v_hh, 'b-', linewidth=2)
    plt.xlabel('时间 (ms)')
    plt.ylabel('膜电位 (mV)')
    plt.title('Hodgkin-Huxley模型')
    plt.grid(True)
    
    plt.subplot(2, 2, 2)
    plt.plot(t, v_izh, 'r-', linewidth=2)
    plt.xlabel('时间 (ms)')
    plt.ylabel('膜电位 (mV)')
    plt.title('Izhikevich模型')
    plt.grid(True)
    
    plt.subplot(2, 2, 3)
    plt.plot(t, I_ext, 'k-', linewidth=2)
    plt.xlabel('时间 (ms)')
    plt.ylabel('电流 (μA/cm²)')
    plt.title('输入电流')
    plt.grid(True)
```

### 2.2 神经网络建模

#### 人工神经网络

**基本结构**：
- 输入层
- 隐藏层
- 输出层

**激活函数**：
- Sigmoid函数
- ReLU函数
- Tanh函数

**学习算法**：
- 反向传播算法
- 梯度下降算法
- 优化算法

#### Python实现
```python
import numpy as np
import matplotlib.pyplot as plt

class ArtificialNeuralNetwork:
    """人工神经网络"""
    
    def __init__(self, layer_sizes, learning_rate=0.01):
        """初始化"""
        self.layer_sizes = layer_sizes
        self.learning_rate = learning_rate
        
        # 初始化权重和偏置
        self.weights = []
        self.biases = []
        
        for i in range(len(layer_sizes) - 1):
            w = np.random.randn(layer_sizes[i], layer_sizes[i+1]) * 0.01
            b = np.zeros((1, layer_sizes[i+1]))
            self.weights.append(w)
            self.biases.append(b)
    
    def sigmoid(self, z):
        """Sigmoid激活函数"""
        return 1 / (1 + np.exp(-np.clip(z, -500, 500)))
    
    def sigmoid_derivative(self, z):
        """Sigmoid导数"""
        s = self.sigmoid(z)
        return s * (1 - s)
    
    def relu(self, z):
        """ReLU激活函数"""
        return np.maximum(0, z)
    
    def relu_derivative(self, z):
        """ReLU导数"""
        return (z > 0).astype(float)
    
    def forward_propagation(self, X):
        """前向传播"""
        self.cache = []
        A = X
        
        for i in range(len(self.weights) - 1):
            Z = np.dot(A, self.weights[i]) + self.biases[i]
            A = self.relu(Z)
            self.cache.append((Z, A))
        
        # 输出层
        Z = np.dot(A, self.weights[-1]) + self.biases[-1]
        A = self.sigmoid(Z)
        self.cache.append((Z, A))
        
        return A
    
    def compute_cost(self, Y, Y_pred):
        """计算损失"""
        m = Y.shape[0]
        cost = -np.sum(Y * np.log(Y_pred + 1e-8) + (1 - Y) * np.log(1 - Y_pred + 1e-8)) / m
        return cost
    
    def backward_propagation(self, X, Y):
        """反向传播"""
        m = X.shape[0]
        Y_pred = self.forward_propagation(X)
        
        # 输出层梯度
        dZ = Y_pred - Y
        dW = np.dot(self.cache[-2][1].T, dZ) / m
        db = np.sum(dZ, axis=0, keepdims=True) / m
        
        self.weights[-1] -= self.learning_rate * dW
        self.biases[-1] -= self.learning_rate * db
        
        # 隐藏层梯度
        for i in range(len(self.weights) - 2, -1, -1):
            dA = np.dot(dZ, self.weights[i+1].T)
            dZ = dA * self.relu_derivative(self.cache[i][0])
            dW = np.dot(self.cache[i-1][1].T, dZ) / m
            db = np.sum(dZ, axis=0, keepdims=True) / m
            
            self.weights[i] -= self.learning_rate * dW
            self.biases[i] -= self.learning_rate * db
    
    def train(self, X, Y, epochs=1000):
        """训练网络"""
        costs = []
        
        for epoch in range(epochs):
            Y_pred = self.forward_propagation(X)
            cost = self.compute_cost(Y, Y_pred)
            costs.append(cost)
            
            if epoch % 100 == 0:
                print(f"Epoch {epoch}, Cost: {cost:.4f}")
            
            self.backward_propagation(X, Y)
        
        return costs

class SpikingNeuralNetwork:
    """脉冲神经网络"""
    
    def __init__(self, layer_sizes, learning_rate=0.01):
        """初始化"""
        self.layer_sizes = layer_sizes
        self.learning_rate = learning_rate
        
        # 初始化权重和偏置
        self.weights = []
        self.biases = []
        
        for i in range(len(layer_sizes) - 1):
            w = np.random.randn(layer_sizes[i], layer_sizes[i+1]) * 0.01
            b = np.zeros((1, layer_sizes[i+1]))
            self.weights.append(w)
            self.biases.append(b)
    
    def leaky_integrate_and_fire(self, v, i, tau=10, v_th=1.0, v_reset=-1.0):
        """Leaky Integrate-and-Fire模型"""
        dv_dt = (-v + i) / tau
        v = v + dv_dt * 0.01
        
        # 检测脉冲
        spike = (v >= v_th).astype(float)
        v = np.where(spike, v_reset, v)
        
        return v, spike
    
    def forward_propagation(self, X, time_steps=100):
        """前向传播"""
        batch_size = X.shape[0]
        spikes = []
        
        # 初始化膜电位
        v = np.zeros((batch_size, self.layer_sizes[0]))
        
        for t in range(time_steps):
            # 输入层
            if t == 0:
                v = X
            
            # 隐藏层和输出层
            for i in range(len(self.weights)):
                # 计算膜电位
                v = np.dot(v, self.weights[i]) + self.biases[i]
                
                # 脉冲发放
                v, spike = self.leaky_integrate_and_fire(v, v)
                
                # 保存脉冲
                spikes.append(spike)
        
        return spikes
    
    def train(self, X, Y, epochs=100):
        """训练网络"""
        costs = []
        
        for epoch in range(epochs):
            # 前向传播
            spikes = self.forward_propagation(X)
            
            # 计算损失
            Y_pred = np.mean(spikes[-1], axis=0)
            cost = np.mean((Y_pred - Y) ** 2)
            costs.append(cost)
            
            if epoch % 10 == 0:
                print(f"Epoch {epoch}, Cost: {cost:.4f}")
            
            # 简单的权重更新
            for i in range(len(self.weights)):
                # 梯度计算
                grad = np.random.randn(*self.weights[i].shape) * 0.01
                
                # 更新权重
                self.weights[i] -= self.learning_rate * grad
        
        return costs
```

### 2.3 神经形态学建模

#### 神经形态学

**定义**：神经形态学是研究神经元形态结构的学科。

**主要特征**：
- 树突形态
- 轴突形态
- 突触分布
- 树突棘形态

#### Python实现
```python
import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

class NeuronMorphology:
    """神经元形态学"""
    
    def __init__(self, soma_radius=10, dendrite_sections=10):
        """初始化"""
        self.soma_radius = soma_radius
        self.dendrite_sections = dendrite_sections
        self.sections = []
    
    def create_soma(self):
        """创建胞体"""
        theta = np.linspace(0, 2*np.pi, 100)
        x = self.soma_radius * np.cos(theta)
        y = self.soma_radius * np.sin(theta)
        z = np.zeros_like(x)
        
        return np.column_stack([x, y, z])
    
    def create_dendrite(self, length=100, diameter=2, sections=10):
        """创建树突"""
        # 创建分段
        segment_length = length / sections
        x = np.linspace(0, length, sections + 1)
        y = np.random.normal(0, 1, sections + 1)
        z = np.random.normal(0, 1, sections + 1)
        
        # 计算每个分段的直径
        diameters = np.linspace(diameter, diameter*0.5, sections + 1)
        
        # 创建每个分段的圆环
        dendrite = []
        for i in range(sections):
            theta = np.linspace(0, 2*np.pi, 20)
            X = np.outer(np.ones_like(theta), np.linspace(x[i], x[i+1]))
            Y = np.outer(np.sin(theta), np.ones(20))
            Z = np.outer(np.cos(theta), np.ones(20))
            
            # 缩放和位移
            scale = diameters[i] / 2
            X = X * scale
            Y = Y * scale
            Z = Z * scale
            
            # 旋转到正确方向
            # 简化的旋转
            X = X + x[i]
            Y = Y + y[i]
            Z = Z + z[i]
            
            dendrite.append(np.column_stack([X.ravel(), Y.ravel(), Z.ravel()]))
        
        return np.vstack(dendrite)
    
    def create_axon(self, length=500, diameter=1, sections=20):
        """创建轴突"""
        # 创建分段
        segment_length = length / sections
        x = np.linspace(0, length, sections + 1)
        y = np.random.normal(0, 0.5, sections + 1)
        z = np.random.normal(0, 0.5, sections + 1)
        
        # 计算每个分段的直径
        diameters = np.linspace(diameter, diameter*0.3, sections + 1)
        
        # 创建每个分段的圆环
        axon = []
        for i in range(sections):
            theta = np.linspace(0, 2*np.pi, 15)
            X = np.outer(np.ones_like(theta), np.linspace(x[i], x[i+1]))
            Y = np.outer(np.sin(theta), np.ones(15))
            Z = np.outer(np.cos(theta), np.ones(15))
            
            # 缩放和位移
            scale = diameters[i] / 2
            X = X * scale
            Y = Y * scale
            Z = Z * scale
            
            # 旋转到正确方向
            X = X + x[i]
            Y = Y + y[i]
            Z = Z + z[i]
            
            axon.append(np.column_stack([X.ravel(), Y.ravel(), Z.ravel()]))
        
        return np.vstack(axon)
    
    def visualize_morphology(self):
        """可视化形态学"""
        # 创建胞体
        soma = self.create_soma()
        
        # 创建树突
        dendrite = self.create_dendrite()
        
        # 创建轴突
        axon = self.create_axon()
        
        # 绘制3D图
        fig = plt.figure(figsize=(12, 8))
        ax = fig.add_subplot(111, projection='3d')
        
        # 绘制胞体
        ax.scatter(soma[:, 0], soma[:, 1], soma[:, 2], c='red', s=100, label='胞体')
        
        # 绘制树突
        ax.scatter(dendrite[:, 0], dendrite[:, 1], dendrite[:, 2], c='blue', s=1, alpha=0.6, label='树突')
        
        # 绘制轴突
        ax.scatter(axon[:, 0], axon[:, 1], axon[:, 2], c='green', s=1, alpha=0.6, label='轴突')
        
        ax.set_xlabel('X (μm)')
        ax.set_ylabel('Y (μm)')
        ax.set_zlabel('Z (μm)')
        ax.set_title('神经元形态学')
        ax.legend()
        
        plt.show()
        
        return soma, dendrite, axon

class SynapseDistribution:
    """突触分布"""
    
    def __init__(self, neuron_morphology):
        """初始化"""
        self.neuron = neuron_morphology
    
    def generate_synapse_positions(self, density=0.1):
        """生成突触位置"""
        # 创建树突
        dendrite = self.neuron.create_dendrite()
        
        # 生成随机突触位置
        n_synapses = int(len(dendrite) * density)
        indices = np.random.choice(len(dendrite), n_synapses, replace=False)
        
        synapse_positions = dendrite[indices]
```

```python
class DiseaseModel:
    """疾病模型"""

        """初始化"""

class AlzheimerModel(DiseaseModel):
    """阿尔茨海默病模型"""
    
    def __init__(self):
        """初始化"""
        super().__init__("阿尔茨海默病")
        
        # 模型参数
        self.beta = 0.01  # β-淀粉样蛋白生成率
        self.gamma = 0.001  # β-淀粉样蛋白清除率
        self.tau = 0.0001  # 神经纤维缠结形成率
        self.phi = 0.00001  # 神经纤维缠结清除率
        self.rho = 0.001  # 神经元死亡率
    
    def differential_equations(self, y, t):
        """微分方程"""
        a, n, d = y  # a: β-淀粉样蛋白, n: 神经纤维缠结, d: 神经元
        da_dt = self.beta - self.gamma * a - self.tau * n * a
        dn_dt = self.tau * n * a - self.phi * n
        dd_dt = -self.rho * d
        
        return [da_dt, dn_dt, dd_dt]
    
    def simulate(self, params=None, t_max=100, dt=0.1):
        """模拟"""
        if params is None:
            # 初始条件
            y0 = [0.1, 0.01, 1.0]  # β-淀粉样蛋白, 神经纤维缠结, 神经元
        else:
            y0 = params
        
        # 时间点
        t = np.arange(0, t_max, dt)
        
        # 求解微分方程
        y = odeint(self.differential_equations, y0, t)
        
        return t, y

class ParkinsonModel(DiseaseModel):
    """帕金森病模型"""
    
    def __init__(self):
        """初始化"""
        super().__init__("帕金森病")
        
        # 模型参数
        self.beta = 0.01  # 多巴胺生成率
        self.gamma = 0.001  # 多巴胺清除率
        self.delta = 0.0001  # 神经元死亡率
        self.rho = 0.00001  # 柯萨奇小体形成率
    
    def differential_equations(self, y, t):
        """微分方程"""
        d, n, c = y  # d: 多巴胺, n: 神经元, c: 柯萨奇小体
        dd_dt = self.beta - self.gamma * d - self.rho * c * d
        dn_dt = -self.delta * n
        dc_dt = self.rho * c * d - 0.001 * c
        
        return [dd_dt, dn_dt, dc_dt]
    
    def simulate(self, params=None, t_max=100, dt=0.1):
        """模拟"""
        if params is None:
            # 初始条件
            y0 = [0.1, 1.0, 0.01]  # 多巴胺, 神经元, 柯萨奇小体
        else:
            y0 = params
        
        # 时间点
        t = np.arange(0, t_max, dt)
        
        # 求解微分方程
        y = odeint(self.differential_equations, y0, t)
        
        return t, y

class BrainComputerInterface:
    """脑机接口"""
    
    def __init__(self):
        """初始化"""
        self.signal_processor = SignalProcessor()
        self.decoder = Decoder()
    
    def process_signal(self, neural_signal):
        """处理信号"""
        # 信号预处理
        processed_signal = self.signal_processor.preprocess(neural_signal)
        
        # 特征提取
        features = self.signal_processor.extract_features(processed_signal)
        
        return features
    
    def decode_intention(self, features):
        """解码意图"""
        intention = self.decoder.decode(features)
        return intention

class SignalProcessor:
    """信号处理器"""
    
    def __init__(self):
        """初始化"""
        pass
    
    def preprocess(self, signal):
        """预处理"""
        # 去噪
        filtered_signal = self.remove_noise(signal)
        
        # 归一化
        normalized_signal = self.normalize(filtered_signal)
        
        return normalized_signal
    
    def extract_features(self, signal):
        """特征提取"""
        # 时间域特征
        time_features = self.extract_time_features(signal)
        
        # 频域特征
        freq_features = self.extract_frequency_features(signal)
        
        # 统计特征
        stat_features = self.extract_statistical_features(signal)
        
        return np.concatenate([time_features, freq_features, stat_features])
    
    def remove_noise(self, signal):
        """去噪"""
        # 简单的移动平均滤波
        kernel_size = 5
        kernel = np.ones(kernel_size) / kernel_size
        filtered_signal = np.convolve(signal, kernel, mode='same')
        
        return filtered_signal
    
    def normalize(self, signal):
        """归一化"""
        signal_mean = np.mean(signal)
        signal_std = np.std(signal)
        normalized_signal = (signal - signal_mean) / signal_std
        
        return normalized_signal
    
    def extract_time_features(self, signal):
        """时间域特征"""
        features = np.array([
            np.mean(signal),
            np.std(signal),
            np.max(signal),
            np.min(signal),
            np.median(signal)
        ])
        
        return features
    
    def extract_frequency_features(self, signal):
        """频域特征"""
        # FFT
        fft = np.fft.fft(signal)
        freq = np.fft.fftfreq(len(signal), d=0.001)
        
        # 功率谱密度
        psd = np.abs(fft) ** 2
        
        # 主要频率
        dominant_freq = freq[np.argmax(psd)]
        
        features = np.array([
            dominant_freq,
            np.mean(psd),
            np.max(psd),
            np.sum(psd)
        ])
        
        return features
    
    def extract_statistical_features(self, signal):
        """统计特征"""
        features = np.array([
            np.mean(np.diff(signal)),
            np.std(np.diff(signal)),
            np.max(np.diff(signal)),
            np.min(np.diff(signal))
        ])
        
        return features

class Decoder:
    """解码器"""
    
    def __init__(self):
        """初始化"""
        self.model = None
    
    def decode(self, features):
        """解码"""
        if self.model is None:
            # 简单的线性分类器
            self.model = LinearClassifier()
        
        intention = self.model.predict(features)
        return intention

class LinearClassifier:
    """线性分类器"""
    
    def __init__(self):
        """初始化"""
        self.weights = np.random.randn(10, 3)  # 10个特征，3个类别
        self.bias = np.zeros(3)
    
    def predict(self, features):
        """预测"""
        scores = np.dot(features, self.weights) + self.bias
        intention = np.argmax(scores)
        
        return intention

class NeuromorphicComputing:
    """神经形态学计算"""
    
    def __init__(self):
        """初始化"""
        self.memory = []
        self.learning_rate = 0.01
    
    def memristor_update(self, memristor_state, current):
        """忆阻器更新"""
        # 简化的忆阻器模型
        delta_state = self.learning_rate * current * (1 - memristor_state)
        new_state = memristor_state + delta_state
        
        return np.clip(new_state, 0, 1)
    
    def synapse_update(self, synapse_weight, pre_spike, post_spike):
        """突触更新"""
        # STDP规则
        if post_spike > 0:
            # 后发放时序依赖可塑性
            delta_weight = self.learning_rate * pre_spike * post_spike
        elif pre_spike > 0:
            # 前发放时序依赖可塑性
            delta_weight = -self.learning_rate * pre_spike * post_spike
        
        new_weight = synapse_weight + delta_weight
        
        return np.clip(new_weight, -1, 1)
    
    def neural_network_forward(self, inputs, weights, biases):
        """神经网络前向传播"""
        # 简化的神经网络
        activations = []
        current = inputs
        
        for i in range(len(weights)):
            # 矩阵乘法
            z = np.dot(current, weights[i]) + biases[i]
            
            # 激活函数
            a = 1 / (1 + np.exp(-np.clip(z, -500, 500)))
            
            activations.append(a)
            current = a
        
        return current, activations
```
