**最终代码（滚动数组）：**
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        if (n == 0) return 0;
        
        int cash = 0;           // 不持有股票的最大利润
        int hold = -prices[0];  // 持有股票的最大利润
        
        for (int i = 1; i < n; ++i) {
            int new_cash = max(cash, hold + prices[i]);  // 不动 / 卖出
            int new_hold = max(hold, -prices[i]);         // 不动 / 买入
            
            cash = new_cash;
            hold = new_hold;
        }
        
        return cash;
    }
};
```

**最终代码（贪心）：**
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        if (n < 2)
            return 0;
        int profit = 0;
        for (int i = 1; i < n; ++i) {
            if (prices[i] > prices[i - 1]) {
                profit += prices[i] - prices[i - 1];
            }
        }
        return profit;
    }
};
```
