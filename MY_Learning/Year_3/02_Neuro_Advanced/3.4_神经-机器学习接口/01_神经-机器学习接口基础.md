# 3.4 神经-机器学习接口

> **专业视角**：计算神经科学与机器学习正在双向融合。神经科学为AI提供生物启发的架构与学习规则；机器学习为神经科学提供分析大规模神经数据的工具。本模块覆盖神经网络与生物神经网络的对应关系、脉冲神经网络的机器学习应用、以及神经数据的机器学习分析方法。

## 4.1 生物神经网络与人工神经网络的对应

### 结构对比

| 生物神经网络 | 人工神经网络 | 对应关系 |
|-------------|-------------|---------|
| 神经元 | 节点/神经元 | 输入加权求和 + 非线性激活 |
| 突触 | 权重 | 可学习的连接强度 |
| 树突 | 输入加权 | 权重乘积累加 |
| 轴突 | 输出层 | 激活值传播 |
| 神经递质 | 激活函数类型 | 决定信号传递特性 |
| 可塑性 | 梯度下降 | 权重随经验调整 |

### 关键差异

| 特性 | 生物神经网络 | 人工神经网络 |
|------|-------------|-------------|
| 时序编码 | 脉冲时序 | 静态向量 |
| 功耗 | ~20W | 大模型数百kW |
| 学习方式 | STDP/Hebbian | 反向传播 |
| 噪声 | 高（随机离子通道）| 低（确定性计算）|
| 容错性 | 极高 | 中等 |

## 4.2 脉冲神经网络（Spiking Neural Networks）

### LIF神经元模型

```python
class LIFNeuron:
    """Leaky Integrate-and-Fire神经元"""

    def __init__(self, tau=20.0, v_thresh=-50.0, v_rest=-70.0, v_reset=-75.0):
        self.tau = tau          # 时间常数(ms)
        self.v_thresh = v_thresh
        self.v_rest = v_rest
        self.v_reset = v_reset
        self.v = v_rest         # 初始膜电位

    def step(self, I, dt=1.0):
        """
        更新膜电位
        I: 输入电流(pA)
        dt: 时间步长(ms)
        """
        # 膜电位微分方程
        dv = (-(self.v - self.v_rest) + I * self.tau) / self.tau
        self.v += dv * dt

        # 阈值检测与重置
        if self.v >= self.v_thresh:
            spike = 1.0
            self.v = self.v_reset
        else:
            spike = 0.0

        return spike
```

### HH神经元模型（Hodgkin-Huxley）

```python
class HHNeuron:
    """Hodgkin-Huxley神经元模型"""

    def __init__(self, C=1.0, g_Na=120.0, g_K=36.0, g_L=0.3,
                 E_Na=50.0, E_K=-77.0, E_L=-54.4):
        self.C = C          # 膜电容(μF/cm²)
        self.g = {'Na': g_Na, 'K': g_K, 'L': g_L}
        self.E = {'Na': E_Na, 'K': E_K, 'L': E_L}
        self.v = -65.0      # 初始膜电位(mV)

    def alpha_m(self, v):
        return 0.1 * (v + 40.0) / (1.0 - np.exp(-(v + 40.0) / 10.0))

    def m_infinity(self, v):
        """Na通道激活门稳态值"""
        return self.alpha_m(v) / (self.alpha_m(v) + self.beta_m(v))

    def step(self, I, dt=0.1):
        """单步更新"""
        # 门变量稳态值
        m = self.m_infinity(self.v)
        h = self.h_infinity(self.v)
        n = self.n_infinity(self.v)

        # 离子电流
        I_Na = self.g['Na'] * m**3 * h * (self.v - self.E['Na'])
        I_K = self.g['K'] * n**4 * (self.v - self.E['K'])
        I_L = self.g['L'] * (self.v - self.E['L'])

        # 膜电位更新
        dv = (-(I_Na + I_K + I_L) + I) / self.C
        self.v += dv * dt

        return 1.0 if self.v > 0 else 0.0
```

## 4.3 神经编码与解码

### 常用编码方式

```python
class NeuralEncoding:
    """神经编码方法集合"""

    @staticmethod
    def rate_coding(spike_train, window=50.0):
        """频率编码：单位时间内的脉冲数"""
        return len(spike_train) / (window / 1000.0)  # spikes per second

    @staticmethod
    def temporal_coding(spike_train):
        """
        时序编码：精确的脉冲时间
        返回：相邻脉冲间隔(ISIs)
        """
        if len(spike_train) < 2:
            return np.array([])
        return np.diff(spike_train)

    @staticmethod
    def population_vector(stable_activity, preferred_angles):
        """
        群体向量编码：用于方向选择性神经元
        应用：运动方向解码
        """
        return np.sum(stable_activity[:, None] * preferred_angles[None, :], axis=0)

    @staticmethod
    def spike_trigger_average(stimulus, spike_times, window=(-50, 200)):
        """
        脉冲触发平均（STA）
        用于估算神经元的感受野
        """
        n_spikes = len(spike_times)
        n_timepoints = int((window[1] - window[0]))

        if n_spikes == 0:
            return np.zeros(n_timepoints)

        sta = np.zeros(n_timepoints)
        for t in spike_times:
            start_idx = int(t + window[0])
            end_idx = int(t + window[1])
            if start_idx >= 0 and end_idx <= len(stimulus):
                sta += stimulus[start_idx:end_idx]

        return sta / n_spikes
```

### Kalman Filter神经解码

```python
class KalmanFilter:
    """Kalman滤波器：最优线性解码"""

    def __init__(self, A, H, Q, R, x0, P0):
        """
        A: 状态转移矩阵
        H: 观测矩阵
        Q: 过程噪声协方差
        R: 观测噪声协方差
        x0: 初始状态
        P0: 初始协方差
        """
        self.A = A
        self.H = H
        self.Q = Q
        self.R = R
        self.x = x0
        self.P = P0

    def predict(self):
        self.x = self.A @ self.x
        self.P = self.A @ self.P @ self.A.T + self.Q
        return self.x

    def update(self, z):
        """观测更新"""
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)  # Kalman增益
        self.x = self.x + K @ (z - self.H @ self.x)
        self.P = (np.eye(len(self.P)) - K @ self.H) @ self.P
        return self.x

    def decode(self, neural_activity, population_matrix):
        """
        解码神经活动
        neural_activity: 神经元发放率
        population_matrix: 神经元的调谐曲线
        """
        self.H = population_matrix
        return self.update(neural_activity)
```

## 4.4 机器学习分析神经数据

### 降维与可视化

```python
from sklearn.decomposition import PCA
from sklearn.manifold import TSNE
import numpy as np

def dimensionality_reduction(neural_data, method='pca', perplexity=30):
    """
    神经数据降维
    neural_data: (n_timepoints, n_neurons) 矩阵
    """
    # PCA降维
    pca = PCA(n_components=min(50, neural_data.shape))
    pca_result = pca.fit_transform(neural_data.T).T  # 转置后对神经元降维

    # t-SNE可视化
    tsne = TSNE(n_components=2, perplexity=perplexity, random_state=42)
    tsne_result = tsne.fit_transform(neural_data)

    return pca_result, tsne_result
```

### 神经数据分类

```python
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis as LDA
from sklearn.model_selection import cross_val_score
from sklearn.svm import SVC

def classify_stimuli(neural_responses, labels, method='lda'):
    """
    神经数据分类：识别刺激类型
    neural_responses: (n_trials, n_neurons)
    labels: (n_trials,)
    """
    if method == 'lda':
        clf = LDA()
    else:
        clf = SVC(kernel='linear')

    scores = cross_val_score(clf, neural_responses, labels, cv=5)
    return scores.mean(), scores.std()
```

### GLM（广义线性模型）神经元建模

```python
import statsmodels.api as sm

class GLMNeuron:
    """基于GLM的神经元模型"""

    def __init__(self, basis_size=10):
        self.basis_size = basis_size
        self.coeffs = None

    def design_matrix(self, spike_history, stimulus_history):
        """
        构建设计矩阵
        - 历史脉冲的平滑版本（神经元的不应期）
        - 视觉刺激的线性滤波
        """
        # 脉冲历史特征
        from scipy.ndimage import gaussian_filter1d
        spike_features = gaussian_filter1d(
            spike_history.astype(float),
            sigma=3
        )

        # 刺激历史特征（延迟通道）
        stimulus_features = np.column_stack([
            np.roll(stimulus_history, i)
            for i in range(self.basis_size)
        ])

        return np.column_stack([spike_features, stimulus_features])

    def fit(self, spike_train, stimulus):
        """拟合GLM"""
        X = self.design_matrix(spike_train, stimulus)
        X = sm.add_constant(X)

        # 使用泊松回归
        model = sm.GLM(spike_train, X, family=sm.families.Poisson())
        self.coeffs = model.fit().params

        return self

    def predict(self, spike_train, stimulus):
        """预测发放率"""
        X = self.design_matrix(spike_train, stimulus)
        X = sm.add_constant(X)
        return np.exp(X @ self.coeffs)
```
