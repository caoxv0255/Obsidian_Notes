## 10.1 栈（Stack）

栈是一种后进先出（LIFO）的数据结构。

### C++实现（使用std::vector）

```cpp
#include <vector>
#include <stdexcept>

template <typename T>
class Stack {
private:
    std::vector<T> data_;

public:
    // 压栈
    void push(const T& value) {
        data_.push_back(value);
    }
    
    void push(T&& value) {
        data_.push_back(std::move(value));
    }
    
    // 弹栈
    T pop() {
        if (empty()) {
            throw std::out_of_range("Stack is empty");
        }
        T value = std::move(data_.back());
        data_.pop_back();
        return value;
    }
    
    // 查看栈顶元素
    T& top() {
        if (empty()) {
            throw std::out_of_range("Stack is empty");
        }
        return data_.back();
    }
    
    const T& top() const {
        if (empty()) {
            throw std::out_of_range("Stack is empty");
        }
        return data_.back();
    }
    
    // 检查是否为空
    bool empty() const noexcept {
        return data_.empty();
    }
    
    // 栈的大小
    size_t size() const noexcept {
        return data_.size();
    }
};

// ============ 测试用例 ============

void test_stack() {
    std::cout << "=== Stack 测试 ===" << std::endl;
    
    Stack<int> stack;
    
    // 压栈操作
    stack.push(1);
    stack.push(2);
    stack.push(3);
    
    std::cout << "Stack size: " << stack.size() << std::endl;
    std::cout << "Top element: " << stack.top() << std::endl;
    
    // 弹栈操作
    while (!stack.empty()) {
        std::cout << "Pop: " << stack.pop() << std::endl;
    }
    
    std::cout << "Stack is empty: " << (stack.empty() ? "true" : "false") << std::endl;
}

// 复杂度分析:
// - push: O(1) 分摊
// - pop: O(1)
// - top: O(1)
// - empty/size: O(1)
```


## 10.2 队列（Queue）

队列是一种先进先出（FIFO）的数据结构。

### C++实现（使用std::deque）

```cpp
#include <deque>
#include <stdexcept>

template <typename T>
class Queue {
private:
    std::deque<T> data_;

public:
    // 入队
    void enqueue(const T& value) {
        data_.push_back(value);
    }
    
    void enqueue(T&& value) {
        data_.push_back(std::move(value));
    }
    
    // 出队
    T dequeue() {
        if (empty()) {
            throw std::out_of_range("Queue is empty");
        }
        T value = std::move(data_.front());
        data_.pop_front();
        return value;
    }
    
    // 查看队首元素
    T& front() {
        if (empty()) {
            throw std::out_of_range("Queue is empty");
        }
        return data_.front();
    }
    
    const T& front() const {
        if (empty()) {
            throw std::out_of_range("Queue is empty");
        }
        return data_.front();
    }
    
    // 检查是否为空
    bool empty() const noexcept {
        return data_.empty();
    }
    
    // 队列的大小
    size_t size() const noexcept {
        return data_.size();
    }
};

// ============ 测试用例 ============

void test_queue() {
    std::cout << "=== Queue 测试 ===" << std::endl;
    
    Queue<int> queue;
    
    // 入队操作
    queue.enqueue(1);
    queue.enqueue(2);
    queue.enqueue(3);
    
    std::cout << "Queue size: " << queue.size() << std::endl;
    std::cout << "Front element: " << queue.front() << std::endl;
    
    // 出队操作
    while (!queue.empty()) {
        std::cout << "Dequeue: " << queue.dequeue() << std::endl;
    }
    
    std::cout << "Queue is empty: " << (queue.empty() ? "true" : "false") << std::endl;
}

// 复杂度分析:
// - enqueue: O(1) 分摊
// - dequeue: O(1)
// - front: O(1)
// - empty/size: O(1)
```


## 10.3 链表（Linked List）

### 单链表

```cpp
#include <memory>

template <typename T>
class SinglyLinkedList {
private:
    struct Node {
        T data;
        std::unique_ptr<Node> next;
        
        Node(const T& value) : data(value), next(nullptr) {}
        Node(T&& value) : data(std::move(value)), next(nullptr) {}
    };
    
    std::unique_ptr<Node> head_;
    size_t size_;

public:
    SinglyLinkedList() : head_(nullptr), size_(0) {}
    
    // 在头部插入
    void push_front(const T& value) {
        auto new_node = std::make_unique<Node>(value);
        new_node->next = std::move(head_);
        head_ = std::move(new_node);
        ++size_;
    }
    
    // 在头部删除
    void pop_front() {
        if (head_) {
            head_ = std::move(head_->next);
            --size_;
        }
    }
    
    // 查找元素
    std::optional<size_t> find(const T& value) const {
        Node* current = head_.get();
        size_t index = 0;
        
        while (current) {
            if (current->data == value) {
                return index;
            }
            current = current->next.get();
            ++index;
        }
        
        return std::nullopt;
    }
    
    // 反转链表
    void reverse() {
        std::unique_ptr<Node> prev = nullptr;
        std::unique_ptr<Node> current = std::move(head_);
        
        while (current) {
            auto next = std::move(current->next);
            current->next = std::move(prev);
            prev = std::move(current);
            current = std::move(next);
        }
        
        head_ = std::move(prev);
    }
    
    bool empty() const noexcept { return head_ == nullptr; }
    size_t size() const noexcept { return size_; }
};

// 复杂度分析:
// - push_front: O(1)
// - pop_front: O(1)
// - find: O(n)
// - reverse: O(n)
```

### 双链表

```cpp
template <typename T>
class DoublyLinkedList {
private:
    struct Node {
        T data;
        Node* prev;
        Node* next;
        
        Node(const T& value) : data(value), prev(nullptr), next(nullptr) {}
    };
    
    Node* head_;
    Node* tail_;
    size_t size_;

public:
    DoublyLinkedList() : head_(nullptr), tail_(nullptr), size_(0) {}
    
    ~DoublyLinkedList() {
        clear();
    }
    
    // 在头部插入
    void push_front(const T& value) {
        auto new_node = new Node(value);
        
        if (head_) {
            new_node->next = head_;
            head_->prev = new_node;
        } else {
            tail_ = new_node;
        }
        
        head_ = new_node;
        ++size_;
    }
    
    // 在尾部插入
    void push_back(const T& value) {
        auto new_node = new Node(value);
        
        if (tail_) {
            new_node->prev = tail_;
            tail_->next = new_node;
        } else {
            head_ = new_node;
        }
        
        tail_ = new_node;
        ++size_;
    }
    
    void clear() {
        while (head_) {
            auto temp = head_;
            head_ = head_->next;
            delete temp;
        }
        tail_ = nullptr;
        size_ = 0;
    }
    
    bool empty() const noexcept { return head_ == nullptr; }
    size_t size() const noexcept { return size_; }
};

// 复杂度分析:
// - push_front/push_back: O(1)
// - pop_front/pop_back: O(1)
// - find: O(n)
```


## 10.4 指针和对象的实现

在C++中，我们可以使用智能指针来安全地管理对象的生命周期。

### 使用std::unique_ptr

```cpp
#include <memory>

class TreeNode {
public:
    int value;
    std::unique_ptr<TreeNode> left;
    std::unique_ptr<TreeNode> right;
    
    TreeNode(int val) : value(val), left(nullptr), right(nullptr) {}
};

// 使用示例
void example_unique_ptr() {
    auto root = std::make_unique<TreeNode>(10);
    root->left = std::make_unique<TreeNode>(5);
    root->right = std::make_unique<TreeNode>(15);
    
    // 当root离开作用域时，所有节点会自动被删除
}

// 优势:
// - 自动内存管理，避免内存泄漏
// - 独占所有权，明确对象的所有者
// - 零开销抽象
```

### 使用std::shared_ptr

```cpp
#include <memory>
#include <vector>

class GraphNode {
public:
    int id;
    std::vector<std::shared_ptr<GraphNode>> neighbors;
    
    explicit GraphNode(int i) : id(i) {}
};

// 使用示例
void example_shared_ptr() {
    auto node1 = std::make_shared<GraphNode>(1);
    auto node2 = std::make_shared<GraphNode>(2);
    
    node1->neighbors.push_back(node2);
    node2->neighbors.push_back(node1);  // 共享所有权
    
    // 两个节点相互引用，引用计数为2
    // 当两个智能指针都被销毁时，节点才会被删除
}

// 注意: 避免循环引用，可以使用std::weak_ptr
```


## 10.5 有根树的表示

### 使用父子指针表示

```cpp
#include <vector>
#include <memory>

template <typename T>
class TreeNode {
public:
    T data;
    TreeNode* parent;
    std::vector<std::unique_ptr<TreeNode>> children;
    
    explicit TreeNode(const T& value) 
        : data(value), parent(nullptr) {}
    
    void add_child(std::unique_ptr<TreeNode> child) {
        child->parent = this;
        children.push_back(std::move(child));
    }
    
    // 深度优先遍历
    void dfs() const {
        std::cout << data << " ";
        for (const auto& child : children) {
            child->dfs();
        }
    }
    
    // 广度优先遍历
    void bfs() const {
        std::queue<const TreeNode*> q;
        q.push(this);
        
        while (!q.empty()) {
            const TreeNode* node = q.front();
            q.pop();
            
            std::cout << node->data << " ";
            
            for (const auto& child : node->children) {
                q.push(child.get());
            }
        }
    }
};

// 使用示例
void test_tree() {
    auto root = std::make_unique<TreeNode<int>(1);
    auto child1 = std::make_unique<TreeNode<int>>(2);
    auto child2 = std::make_unique<TreeNode<int>>(3);
    auto grandchild1 = std::make_unique<TreeNode<int>>(4);
    
    child1->add_child(std::move(grandchild1));
    root->add_child(std::move(child1));
    root->add_child(std::move(child2));
    
    std::cout << "DFS: ";
    root->dfs();
    std::cout << std::endl;
    
    std::cout << "BFS: ";
    root->bfs();
    std::cout << std::endl;
}
```


## 10.6 二叉树

### 二叉树节点定义

```cpp
template <typename T>
class BinaryTreeNode {
public:
    T data;
    std::unique_ptr<BinaryTreeNode> left;
    std::unique_ptr<BinaryTreeNode> right;
    
    explicit BinaryTreeNode(const T& value) 
        : data(value), left(nullptr), right(nullptr) {}
    
    // 前序遍历：根-左-右
    void preorder() const {
        std::cout << data << " ";
        if (left) left->preorder();
        if (right) right->preorder();
    }
    
    // 中序遍历：左-根-右
    void inorder() const {
        if (left) left->inorder();
        std::cout << data << " ";
        if (right) right->inorder();
    }
    
    // 后序遍历：左-右-根
    void postorder() const {
        if (left) left->postorder();
        if (right) right->postorder();
        std::cout << data << " ";
    }
    
    // 层次遍历（BFS）
    void level_order() const {
        std::queue<const BinaryTreeNode*> q;
        q.push(this);
        
        while (!q.empty()) {
            const BinaryTreeNode* node = q.front();
            q.pop();
            
            std::cout << node->data << " ";
            
            if (node->left) q.push(node->left.get());
            if (node->right) q.push(node->right.get());
        }
    }
};

// 使用示例
void test_binary_tree() {
    auto root = std::make_unique<BinaryTreeNode<int>>(1);
    root->left = std::make_unique<BinaryTreeNode<int>>(2);
    root->right = std::make_unique<BinaryTreeNode<int>>(3);
    root->left->left = std::make_unique<BinaryTreeNode<int>>(4);
    root->left->right = std::make_unique<BinaryTreeNode<int>>(5);
    
    std::cout << "Preorder: ";
    root->preorder();
    std::cout << std::endl;
    
    std::cout << "Inorder: ";
    root->inorder();
    std::cout << std::endl;
    
    std::cout << "Postorder: ";
    root->postorder();
    std::cout << std::endl;
    
    std::cout << "Level order: ";
    root->level_order();
    std::cout << std::endl;
}

// 遍历复杂度分析:
// - 时间复杂度: O(n)，n为节点数
// - 空间复杂度: O(h)，h为树的高度（递归栈空间）
```
