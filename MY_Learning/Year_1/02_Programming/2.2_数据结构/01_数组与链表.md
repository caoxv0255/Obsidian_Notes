## 1. 数组（Array）

### 1.1 数组的定义

**数组**（Array）是一种线性数据结构，用于存储相同类型的数据元素，这些元素在内存中连续存储。

### 1.2 数组的特点

- **优点**：
  - 随机访问：可以通过索引直接访问任意元素
  - 内存效率高：元素在内存中连续存储
  - 实现简单

- **缺点**：
  - 大小固定：创建后大小不可改变
  - 插入/删除效率低：需要移动大量元素
  - 预分配内存：可能浪费空间

### 1.3 数组的实现

```python
class Array:
    """数组类"""
    
    def __init__(self, size=10):
        """初始化数组"""
        self.size = size
        self.data = [None] * size
        self.length = 0
    
    def __str__(self):
        """字符串表示"""
        return str(self.data[:self.length])
    
    def __len__(self):
        """数组长度"""
        return self.length
    
    def __getitem__(self, index):
        """访问元素"""
        if 0 <= index < self.length:
            return self.data[index]
        raise IndexError("索引超出范围")
    
    def __setitem__(self, index, value):
        """设置元素"""
        if 0 <= index < self.length:
            self.data[index] = value
        else:
            raise IndexError("索引超出范围")
    
    def append(self, value):
        """添加元素"""
        if self.length < self.size:
            self.data[self.length] = value
            self.length += 1
        else:
            raise OverflowError("数组已满")
    
    def insert(self, index, value):
        """在指定位置插入元素"""
        if 0 <= index <= self.length:
            if self.length < self.size:
                # 移动元素
                for i in range(self.length, index, -1):
                    self.data[i] = self.data[i - 1]
                self.data[index] = value
                self.length += 1
            else:
                raise OverflowError("数组已满")
        else:
            raise IndexError("索引超出范围")
    
    def delete(self, index):
        """删除指定位置的元素"""
        if 0 <= index < self.length:
            # 移动元素
            for i in range(index, self.length - 1):
                self.data[i] = self.data[i + 1]
            self.length -= 1
        else:
            raise IndexError("索引超出范围")
    
    def search(self, value):
        """搜索元素"""
        for i in range(self.length):
            if self.data[i] == value:
                return i
        return -1
    
    def reverse(self):
        """反转数组"""
        for i in range(self.length // 2):
            self.data[i], self.data[self.length - 1 - i] = \
                self.data[self.length - 1 - i], self.data[i]
    
    def sort(self):
        """排序数组"""
        self.data[:self.length] = sorted(self.data[:self.length])

# 使用示例
```

### 1.4 数组的应用

```python
def array_applications():
    """数组应用示例"""
    
    # 1. 矩阵运算
    def matrix_multiplication(A, B):
        """矩阵乘法"""
        m, n = len(A), len(A[0])
        n, p = len(B), len(B[0])
        
        C = [[0] * p for _ in range(m)]
        
        for i in range(m):
            for j in range(p):
                for k in range(n):
                    C[i][j] += A[i][k] * B[k][j]
        
        return C
    
    # 2. 数组排序
    def bubble_sort(arr):
        """冒泡排序"""
        n = len(arr)
        for i in range(n):
            for j in range(0, n - i - 1):
                if arr[j] > arr[j + 1]:
                    arr[j], arr[j + 1] = arr[j + 1], arr[j]
        return arr
    
    # 3. 数组去重
    def remove_duplicates(arr):
        """移除重复元素"""
        seen = set()
        result = []
        for item in arr:
            if item not in seen:
                seen.add(item)
                result.append(item)
        return result
    
    # 测试
    print("=== 数组应用示例 ===")
    
    # 矩阵乘法
    A = [[1, 2], [3, 4]]
    B = [[5, 6], [7, 8]]
    C = matrix_multiplication(A, B)
    print(f"矩阵乘法结果: {C}")
    
    # 排序
    arr = [64, 34, 25, 12, 22, 11, 90]
    sorted_arr = bubble_sort(arr.copy())
    print(f"排序结果: {sorted_arr}")
    
    # 去重
    arr_with_duplicates = [1, 2, 2, 3, 4, 4, 5]
    unique_arr = remove_duplicates(arr_with_duplicates)
    print(f"去重结果: {unique_arr}")
```

## 2. 链表（Linked List）

### 2.1 链表的定义

**链表**（Linked List）是一种线性数据结构，由一系列节点组成，每个节点包含数据和指向下一个节点的指针。

### 2.2 链表的特点

- **优点**：
  - 动态大小：可以动态增长和缩小
  - 插入/删除效率高：只需要修改指针
  - 无需预分配内存

- **缺点**：
  - 不能随机访问：需要从头开始遍历
  - 额外内存开销：存储指针
  - 缓存性能差：节点在内存中不连续

### 2.3 链表的实现

```python
class Node:
    """链表节点"""
    
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    """单链表类"""
    
    def __init__(self):
        """初始化链表"""
        self.head = None
        self.size = 0
    
    def __str__(self):
        """字符串表示"""
        result = []
        current = self.head
        while current:
            result.append(str(current.data))
            current = current.next
        return " -> ".join(result) if result else "空链表"
    
    def __len__(self):
        """链表长度"""
        return self.size
    
    def is_empty(self):
        """检查链表是否为空"""
        return self.head is None
    
    def append(self, data):
        """在链表末尾添加节点"""
        new_node = Node(data)
        if self.is_empty():
            self.head = new_node
        else:
            current = self.head
            while current.next:
                current = current.next
            current.next = new_node
        self.size += 1
    
    def prepend(self, data):
        """在链表开头添加节点"""
        new_node = Node(data)
        new_node.next = self.head
        self.head = new_node
        self.size += 1
    
    def insert(self, index, data):
        """在指定位置插入节点"""
        if index < 0 or index > self.size:
            raise IndexError("索引超出范围")
        
        if index == 0:
            self.prepend(data)
            return
        
        new_node = Node(data)
        current = self.head
        for _ in range(index - 1):
            current = current.next
        
        new_node.next = current.next
        current.next = new_node
        self.size += 1
    
    def delete(self, data):
        """删除指定数据的节点"""
        if self.is_empty():
            return False
        
        # 删除头节点
        if self.head.data == data:
            self.head = self.head.next
            self.size -= 1
            return True
        
        current = self.head
        while current.next:
            if current.next.data == data:
                current.next = current.next.next
                self.size -= 1
                return True
            current = current.next
        
        return False
    
    def delete_at_index(self, index):
        """删除指定位置的节点"""
        if index < 0 or index >= self.size:
            raise IndexError("索引超出范围")
        
        if index == 0:
            self.head = self.head.next
            self.size -= 1
            return
        
        current = self.head
        for _ in range(index - 1):
            current = current.next
        
        current.next = current.next.next
        self.size -= 1
    
    def search(self, data):
        """搜索节点"""
        current = self.head
        index = 0
        while current:
            if current.data == data:
                return index
            current = current.next
            index += 1
        return -1
    
    def get(self, index):
        """获取指定位置的节点数据"""
        if index < 0 or index >= self.size:
            raise IndexError("索引超出范围")
        
        current = self.head
        for _ in range(index):
            current = current.next
        
        return current.data
    
    def reverse(self):
        """反转链表"""
        prev = None
        current = self.head
        
        while current:
            next_node = current.next
            current.next = prev
            prev = current
            current = next_node
        
        self.head = prev
    
    def find_middle(self):
        """找到链表的中间节点"""
        slow = fast = self.head
        
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
        
        return slow.data if slow else None
    
    def has_cycle(self):
        """检测链表是否有环"""
        slow = fast = self.head
        
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            
            if slow == fast:
                return True
        
        return False

# 使用示例
if __name__ == "__main__":
    # 创建链表
    ll = LinkedList()
    
    # 添加元素
    ll.append(1)
    ll.append(2)
    ll.append(3)
    ll.append(4)
    ll.append(5)
    
    print(f"链表: {ll}")
    print(f"链表长度: {len(ll)}")
    
    # 在开头添加
    ll.prepend(0)
    print(f"在开头添加0后: {ll}")
    
    # 在指定位置插入
    ll.insert(2, 10)
    print(f"在位置2插入10后: {ll}")
    
    # 搜索
    index = ll.search(3)
    print(f"3的索引: {index}")
    
    # 获取指定位置的元素
    element = ll.get(2)
    print(f"位置2的元素: {element}")
    
    # 反转链表
    ll.reverse()
    print(f"反转后: {ll}")
    
    # 找到中间节点
    middle = ll.find_middle()
    print(f"中间节点: {middle}")
    
    # 删除节点
    ll.delete(10)
    print(f"删除10后: {ll}")
    
    # 删除指定位置的节点
    ll.delete_at_index(0)
    print(f"删除位置0后: {ll}")
```

### 2.4 双向链表

```python
class DoublyNode:
    """双向链表节点"""
    
    def __init__(self, data):
        self.data = data
        self.next = None
        self.prev = None

class DoublyLinkedList:
    """双向链表类"""
    
    def __init__(self):
        self.head = None
        self.tail = None
        self.size = 0
    
    def __str__(self):
        """字符串表示"""
        result = []
        current = self.head
        while current:
            result.append(str(current.data))
            current = current.next
        return " <-> ".join(result) if result else "空双向链表"
    
    def append(self, data):
        """在链表末尾添加节点"""
        new_node = DoublyNode(data)
        if self.is_empty():
            self.head = self.tail = new_node
        else:
            new_node.prev = self.tail
            self.tail.next = new_node
            self.tail = new_node
        self.size += 1
    
    def prepend(self, data):
        """在链表开头添加节点"""
        new_node = DoublyNode(data)
        if self.is_empty():
            self.head = self.tail = new_node
        else:
            new_node.next = self.head
            self.head.prev = new_node
            self.head = new_node
        self.size += 1
    
    def delete(self, data):
        """删除指定数据的节点"""
        current = self.head
        while current:
            if current.data == data:
                if current == self.head:
                    self.head = current.next
                if current == self.tail:
                    self.tail = current.prev
                
                if current.prev:
                    current.prev.next = current.next
                if current.next:
                    current.next.prev = current.prev
                
                self.size -= 1
                return True
            current = current.next
        return False
    
    def is_empty(self):
        """检查链表是否为空"""
        return self.head is None

# 使用示例
if __name__ == "__main__":
    dll = DoublyLinkedList()
    dll.append(1)
    dll.append(2)
    dll.append(3)
    dll.prepend(0)
    print(f"双向链表: {dll}")
    
    dll.delete(2)
    print(f"删除2后: {dll}")
```

### 5.1 栈的实现

```python
class Stack:
    """栈类（基于数组）"""
    
    def __init__(self):
        self.items = []
    
    def push(self, item):
        """入栈"""
        self.items.append(item)
    
    def pop(self):
        """出栈"""
        if self.is_empty():
            raise IndexError("栈为空")
        return self.items.pop()
    
    def peek(self):
        """查看栈顶元素"""
        if self.is_empty():
            raise IndexError("栈为空")
        return self.items[-1]
    
    def is_empty(self):
        """检查栈是否为空"""
        return len(self.items) == 0
    
    def size(self):
        """获取栈的大小"""
        return len(self.items)

# 使用示例
if __name__ == "__main__":
    stack = Stack()
    stack.push(1)
    stack.push(2)
    stack.push(3)
    print(f"栈顶元素: {stack.peek()}")
    print(f"出栈: {stack.pop()}")
    print(f"栈的大小: {stack.size()}")
```

```python
class Queue:
    """队列类（基于数组）"""
    
    def __init__(self):
        self.items = []
    
    def enqueue(self, item):
        """入队"""
        self.items.append(item)
    
    def dequeue(self):
        """出队"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items.pop(0)
    
    def front(self):
        """查看队首元素"""
        if self.is_empty():
            raise IndexError("队列为空")
        return self.items[0]
    
    def is_empty(self):
        """检查队列是否为空"""
        return len(self.items) == 0
    
    def size(self):
        """获取队列的大小"""
        return len(self.items)

# 使用示例
```

## 6. 常见问题

### Q1: 何时使用数组，何时使用链表？

**A**: 需要频繁随机访问时使用数组，需要频繁插入/删除时使用链表。

### Q2: 链表的内存效率如何？

**A**: 链表的内存效率较低，因为每个节点需要额外的指针空间。

### Q3: 如何检测链表是否有环？

**A**: 使用快慢指针法，快指针每次移动两步，慢指针每次移动一步。
