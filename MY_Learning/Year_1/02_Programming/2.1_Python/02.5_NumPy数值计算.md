### 2.3 收益率计算（核心操作）

```python
# 假设 prices 是日价格序列
prices = np.array([100, 101, 102, 101.5, 103, 104, 102.5, 101, 105, 106])

# 简单收益率（Simple Return）
simple_returns = np.diff(prices) / prices[:-1]

# 对数收益率（Log Return）— 量化中更常用
log_returns = np.log(prices[1:]) - np.log(prices[:-1])
# 或者：
log_returns = np.diff(np.log(prices))

# 年化（假设252个交易日）
annual_return = np.mean(log_returns) * 252
annual_vol = np.std(log_returns, ddof=1) * np.sqrt(252)

# 夏普比率（假设无风险利率 2%）
rf = 0.02
sharpe = (annual_return - rf) / annual_vol

print(f"年化收益率: {annual_return:.2%}")
print(f"年化波动率: {annual_vol:.2%}")
print(f"夏普比率: {sharpe:.3f}")
```

```python
# 布林带（Bollinger Bands）
upper = ma20 + 2 * vol20
lower = ma20 - 2 * vol20

# 协方差矩阵（滚动窗口）
def rolling_cov(x, y, window):
    return np.array([np.cov(x[i-window:i], y[i-window:i])[0,1]
                     for i in range(window, min(len(x), len(y))+1)])
```

## 4. 随机模拟与蒙特卡洛

### 4.1 几何布朗运动（GBM）

BSM 模型假设股价服从几何布朗运动：
$$dS_t = \mu S_t dt + \sigma S_t dW_t$$

离散化（Euler-Maruyama 方法）：
$$S_{t+\Delta t} = S_t \exp\left[\left(\mu - \frac{\sigma^2}{2}\right)\Delta t + \sigma \sqrt{\Delta t} Z\right], \quad Z \sim N(0,1)$$

```python
def gbm(S0, mu, sigma, T, dt, n_paths):
    """模拟 n_paths 条 GBM 路径"""
    n_steps = int(T / dt)
    paths = np.zeros((n_paths, n_steps + 1))
    paths[:, 0] = S0
    for t in range(n_steps):
        Z = np.random.standard_normal(n_paths)
        paths[:, t+1] = paths[:, t] * np.exp((mu - 0.5*sigma**2)*dt + sigma*np.sqrt(dt)*Z)
    return paths

# 模拟10000条路径，期限1年，步长1天
paths = gbm(S0=100, mu=0.05, sigma=0.2, T=1, dt=1/252, n_paths=10000)

# 画前10条路径
import matplotlib.pyplot as plt
for i in range(10):
    plt.plot(paths[i])
plt.title('GBM 股价模拟（前10条路径）')
plt.xlabel('交易日')
plt.ylabel('股价')
plt.savefig('gbm_simulation.png', dpi=150)
plt.show()

# 统计
print(f"期末价格均值: {paths[:, -1].mean():.2f}")
print(f"期末价格标准差: {paths[:, -1].std():.2f}")
print(f"跌破90元的比例: {(paths[:, -1] < 90).mean():.2%}")
```

### 4.2 蒙特卡洛期权定价

```python
def mc_option_price(S0, K, T, r, sigma, n_paths, payoff='call'):
    """蒙特卡洛估算期权价格"""
    paths = gbm(S0, mu=r, sigma=sigma, T=T, dt=1/252, n_paths=n_paths)
    ST = paths[:, -1]  # 到期日股价

    if payoff == 'call':
        payoffs = np.maximum(ST - K, 0)
    else:
        payoffs = np.maximum(K - ST, 0)

    price = np.exp(-r * T) * payoffs.mean()
    se = np.exp(-r * T) * payoffs.std() / np.sqrt(n_paths)  # 标准误
    return price, se

# 定价一份行权价105、期限1年的欧式看涨期权
price, se = mc_option_price(S0=100, K=105, T=1, r=0.05, sigma=0.2, n_paths=100000)
print(f"期权价格: {price:.4f} ± {1.96*se:.4f} (95% CI)")

# 对比 Black-Scholes 解析解
from scipy.stats import norm
d1 = (np.log(S0/K) + (r + 0.5*sigma**2)*T) / (sigma*np.sqrt(T))
d2 = d1 - sigma*np.sqrt(T)
bs_price = S0*norm.cdf(d1) - K*np.exp(-r*T)*norm.cdf(d2)
print(f"BS 解析解: {bs_price:.4f}")
```
