### 2.3 神经元的电活动

#### 静息电位

**定义**：
- 神经元在未受刺激时的膜电位
- 通常为-70mV

**产生机制**：
- 钾离子外流
- 钠钾泵作用
- 离子通道选择性

#### 动作电位

**产生过程**：
1. **去极化**：膜电位从-70mV变为0mV
2. **上升相**：膜电位快速上升到+30mV
3. **下降相**：膜电位快速下降回到-70mV
4. **超极化**：膜电位短暂低于-70mV

**条件**：
- 刺激强度达到阈值
- 刺激时间足够长
- 刺激频率适当

#### 神经冲动传导

**方式**：
- 跳跃式传导
- 连续式传导

**速度**：
- 有髓鞘：100m/s
- 无髓鞘：1m/s

import numpy as np
import matplotlib.pyplot as plt
import networkx as nx
from mpl_toolkits.mplot3d import Axes3D

class Neuron:
    """神经元类"""
    
    def __init__(self, soma_area=1000, dendrite_length=10000, axon_length=100000):
        """初始化"""
        self.soma_area = soma_area
        self.dendrite_length = dendrite_length
        self.axon_length = axon_length
        self.threshold = -55  # 阈值电位 (mV)
        self.resting_potential = -70  # 静息电位 (mV)
        self.action_potential = 30  # 动作电位峰值 (mV)
    
    def calculate_capacitance(self):
        """计算膜电容"""
        # 假设单位面积膜电容为1 μF/cm²
        return self.soma_area * 1e-8  # F
    
    def calculate_resistance(self):
        """计算膜电阻"""
        # 假设单位面积膜电阻为10 kΩ·cm²
        return self.soma_area * 10e3  # Ω
    
    def simulate_action_potential(self, current_injection=0.1, time_steps=1000):
        """模拟动作电位"""
        dt = 0.001  # 时间步长 (s)
        membrane_capacitance = self.calculate_capacitance()
        membrane_resistance = self.calculate_resistance()
        
        # 初始条件
        v = np.full(time_steps, self.resting_potential)
        i = np.full(time_steps, current_injection)
        
        # 数值积分 (欧拉方法)
        for t in range(1, time_steps):
            dv_dt = (-(v[t-1] - self.resting_potential) / membrane_resistance + i[t]) / membrane_capacitance
            v[t] = v[t-1] + dv_dt * dt
            
            # 检测动作电位
            if v[t] > self.action_potential:
                v[t] = self.action_potential  # 限制动作电位峰值
        
        return v
    
    def plot_action_potential(self, v, time_points):
        """绘制动作电位"""
        plt.figure(figsize=(10, 6))
        plt.plot(time_points, v, 'b-', linewidth=2)
        plt.axhline(y=self.threshold, color='r', linestyle='--', label='阈值')
        plt.axhline(y=self.resting_potential, color='g', linestyle='--', label='静息电位')
        plt.xlabel('时间 (ms)')
        plt.ylabel('膜电位 (mV)')
        plt.title('神经元动作电位')
        plt.legend()
        plt.grid(True)
        plt.show()

class NeuralNetwork:
    """神经网络类"""
    
    def __init__(self, num_neurons=100):
        """初始化"""
        self.num_neurons = num_neurons
        self.network = nx.erdos_renyi_graph(num_neurons, 0.1)
        self.neuron_types = np.random.choice(['excitatory', 'inhibitory'], num_neurons)
    
    def add_synapse(self, pre_neuron, post_neuron, weight=0.5):
        """添加突触"""
        if self.neuron_types[pre_neuron] == 'excitatory':
            self.network.add_edge(pre_neuron, post_neuron, weight=weight, type='excitatory')
        else:
            self.network.add_edge(pre_neuron, post_neuron, weight=weight, type='inhibitory')
    
    def simulate_network_activity(self, time_steps=1000):
        """模拟网络活动"""
        # 初始化膜电位
        v = np.full(self.num_neurons, -70)
        
        # 模拟活动
        for t in range(time_steps):
            # 计算输入电流
            i_input = np.zeros(self.num_neurons)
            
            # 添加随机输入
            i_input += np.random.normal(0, 0.1, self.num_neurons)
            
            # 计算突触电流
            for pre, post, data in self.network.edges(data=True):
                if v[pre] > -55:  # 前神经元发放动作电位
                    i_input[post] += data['weight'] * (data['type'] == 'excitatory') - data['weight'] * (data['type'] == 'inhibitory')
            
            # 更新膜电位
            v = v + 0.01 * (i_input - (v - (-70)))  # 简化的膜动力学
            
            # 动作电位检测
            v[v > 30] = 30  # 限制动作电位峰值
            
            # 复位
            v[v > 30] = -70
        
        return v
    
    def visualize_network(self):
        """可视化网络"""
        plt.figure(figsize=(10, 8))
        
        # 绘制网络
        pos = nx.spring_layout(self.network, seed=42)
        nx.draw(self.network, pos, node_color='lightblue', node_size=500, with_labels=False)
        
        # 标记兴奋性和抑制性神经元
        excitatory_nodes = [n for n, t in enumerate(self.neuron_types) if t == 'excitatory']
        inhibitory_nodes = [n for n, t in enumerate(self.neuron_types) if t == 'inhibitory']
        
        nx.draw_networkx_nodes(self.network, pos, nodelist=excitatory_nodes, 
                              node_color='red', node_size=500, label='兴奋性')
        nx.draw_networkx_nodes(self.network, pos, nodelist=inhibitory_nodes, 
                              node_color='blue', node_size=500, label='抑制性')
        
        plt.title('神经网络')
        plt.legend()
        plt.show()

class Neurodevelopment:
    """神经发育类"""
    
    def __init__(self, initial_neurons=100):
        """初始化"""
        self.initial_neurons = initial_neurons
        self.neurons = initial_neurons
        self.synapses = 0
        self.age = 0
    
    def simulate_neurogenesis(self, time_steps=100):
        """模拟神经发生"""
        neurogenesis_rate = 0.1  # 每天神经发生率
        neurodegeneration_rate = 0.01  # 每天神经退化率
        
        neuron_counts = [self.neurons]
        
        for t in range(time_steps):
            # 神经发生
            new_neurons = int(self.neurons * neurogenesis_rate)
            
            # 神经退化
            degenerated_neurons = int(self.neurons * neurodegeneration_rate)
            
            # 更新神经元数量
            self.neurons += new_neurons - degenerated_neurons
            self.neurons = max(self.neurons, 0)  # 确保不为负
            
            # 更新突触数量
            self.synapses += new_neurons * 2  # 每个新神经元形成2个突触
            
            neuron_counts.append(self.neurons)
        
        return np.array(neuron_counts)
    
    def simulate_synaptogenesis(self, time_steps=100):
        """模拟突触发生"""
        synaptogenesis_rate = 0.2  # 每天突触发生率
        synapse_elimination_rate = 0.05  # 每天突触消除率
        
        synapse_counts = [self.synapses]
        
        for t in range(time_steps):
            # 突触发生
            new_synapses = int(self.synapses * synaptogenesis_rate)
            
            # 突触消除
            eliminated_synapses = int(self.synapses * synapse_elimination_rate)
            
            # 更新突触数量
            self.synapses += new_synapses - eliminated_synapses
            self.synapses = max(self.synapses, 0)  # 确保不为负
            
            synapse_counts.append(self.synapses)
        
        return np.array(synapse_counts)
    
    def plot_development(self):
        """绘制发育过程"""
        neuron_counts = self.simulate_neurogenesis(100)
        synapse_counts = self.simulate_synaptogenesis(100)
        
        time_points = np.arange(0, 100)

## 9. 常见问题

### Q1: 神经元的基本结构是什么？

**A**: 神经元的基本结构包括胞体（Soma）、树突（Dendrites）、轴突（Axon）和突触（Synapse）。

### Q2: 神经系统的信号传递是如何进行的？

**A**: 神经系统的信号传递通过动作电位在轴突中传导，到达突触前末梢时释放神经递质，递质扩散到突触间隙并与突触后膜受体结合，引起突触后膜电位变化。

### Q3: 神经可塑性是什么？

**A**: 神经可塑性是指神经系统结构和功能的可变性，包括突触强度变化、新突触形成、神经元再生等。
