## 3. 金融数据计算

### 3.1 收益率计算

```python
# 对数收益率（加法性，适合统计建模）
log_returns = np.log(price / price.shift(1))
```

### 3.2 移动窗口计算

```python
# 移动平均（MA）
ma5 = df['close'].rolling(window=5).mean()
ma10 = df['close'].rolling(window=10).mean()
ma20 = df['close'].rolling(window=20).mean()

# 指数移动平均（EMA）
ema12 = df['close'].ewm(span=12, adjust=False).mean()
ema26 = df['close'].ewm(span=26, adjust=False).mean()

# MACD
macd_line = ema12 - ema26

macd_histogram = macd_line - signal_line

# 波动率（滚动标准差）
rolling_vol = df['close'].pct_change().rolling(window=20).std() * np.sqrt(252)

# 历史波动率（parkinson, garrout, yang-zhang）
def parkinson_vol(high, low, window=20):
    """Parkinson波动率"""
    log_hl = np.log(high / low)
    return np.sqrt(log_hl.pow(2).rolling(window).mean() / (4 * np.log(2)) * 252)
```

### 3.3 技术指标

```python
def calculate_bollinger_bands(price, window=20, num_std=2):
    """布林带"""
    ma = price.rolling(window).mean()
    std = price.rolling(window).std()
    upper = ma + num_std * std
    lower = ma - num_std * std
    return upper, ma, lower

def calculate_rsi(price, window=14):
    """RSI相对强弱指标"""
    delta = price.diff()
    gain = delta.where(delta > 0, 0).rolling(window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window).mean()
    rs = gain / loss
    return 100 - (100 / (1 + rs))

def calculate_atr(high, low, close, window=14):
    """ATR平均真实波幅"""
    tr1 = high - low
    tr2 = abs(high - close.shift(1))
    tr3 = abs(low - close.shift(1))
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    return tr.rolling(window).mean()
```

## 4. 统计分析与检验

### 4.1 描述性统计

```python
# 基本统计
returns.describe()

# 偏度和峰度
```

### 4.3 平稳性检验（ADF检验）

```python
from statsmodels.tsa.stattools import adfuller

def adf_test(series, name=''):
    """ADF平稳性检验"""
    result = adfuller(series.dropna())
    print(f'ADF检验结果 - {name}')
    print(f'ADF统计量: {result[0]:.4f}')
    print(f'p-value: {result[1]:.4f}')
    print(f'临界值: {result[4]}')
    return result[1] < 0.05  # True表示平稳

is_stationary = adf_test(log_returns, '对数收益率')
```

```python
def plot_candlestick(ohlc, start=None, end=None):
    """绘制K线图"""
    if start:
        ohlc = ohlc[start:end]
    
    fig, ax = plt.subplots(figsize=(12, 6))
    
    # 涨跌颜色
    up = ohlc[ohlc['close'] >= ohlc['open']]
    down = ohlc[ohlc['close'] < ohlc['open']]
    
    # 绘制
    ax.bar(up.index, up['high'] - up['low'], bottom=up['low'], color='red', width=0.8)
    ax.bar(up.index, up['close'] - up['open'], bottom=up['open'], color='red', width=0.6)
    
    ax.bar(down.index, down['high'] - down['low'], bottom=down['low'], color='green', width=0.8)
    ax.bar(down.index, down['open'] - down['close'], bottom=down['close'], color='green', width=0.6)
```

### 6.2 完整分析流程

```python
class StockAnalyzer:
    def __init__(self, data):
        self.data = data.copy()
        self.data['returns'] = self.data['close'].pct_change()
        self.data['log_returns'] = np.log(self.data['close'] / self.data['close'].shift(1))
    
    def summary_stats(self):
        """收益率统计"""
        stats = pd.DataFrame({
            '日均收益': self.data['returns'].mean(),
            '年化收益': self.data['returns'].mean() * 252,
            '年化波动': self.data['returns'].std() * np.sqrt(252),
            '夏普比率': (self.data['returns'].mean() * 252) / (self.data['returns'].std() * np.sqrt(252)),
            '最大回撤': self.calculate_max_drawdown(),
            '偏度': self.data['returns'].skew(),
            '峰度': self.data['returns'].kurtosis()
        })
        return stats
    
    def calculate_max_drawdown(self):
        """最大回撤"""
        cumulative = (1 + self.data['returns']).cumprod()
        running_max = cumulative.expanding().max()
        drawdown = (cumulative - running_max) / running_max
        return drawdown.min()
    
    def calculate_sharpe(self, risk_free=0.03):
        """夏普比率"""
        excess_returns = self.data['returns'] - risk_free / 252
        return np.sqrt(252) * excess_returns.mean() / excess_returns.std()
    
    def calculate_sortino(self, risk_free=0.03, target=0):
        """索提诺比率"""
        excess_returns = self.data['returns'] - risk_free / 252
        downside_returns = excess_returns[excess_returns < target]
        return np.sqrt(252) * excess_returns.mean() / downside_returns.std()
```
