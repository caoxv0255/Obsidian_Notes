## 1. 研究实践的定义

### 1.1 研究实践的定义

import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from datetime import datetime, timedelta
import numpy as np

class ProjectManagement:
    """项目管理"""
    
    def __init__(self):
        """初始化"""
        pass
    
    def create_gantt_chart(self, tasks, start_date="2023-01-01"):
        """创建甘特图"""
        fig, ax = plt.subplots(figsize=(12, 8))
        
        # 转换开始日期
        start = datetime.strptime(start_date, "%Y-%m-%d")
        
        # 任务颜色
        colors = plt.cm.Set3(np.linspace(0, 1, len(tasks)))
        
        # 绘制任务条
        for i, task in enumerate(tasks):
            start_time = start + timedelta(days=task['start'])
            end_time = start + timedelta(days=task['start'] + task['duration'])
            
            ax.barh(i, task['duration'], left=task['start'], 
                   height=0.5, color=colors[i], alpha=0.8, label=task['name'])
            
            # 添加任务名称
            ax.text(task['start'] + task['duration']/2, i, task['name'],
                   ha='center', va='center', fontsize=10)
        
        # 设置时间轴
        ax.set_xlabel('时间')
        ax.set_ylabel('任务')
        ax.set_title('项目甘特图')
        
        # 设置x轴格式
        ax.xaxis.set_major_formatter(mdates.DateFormatter('%m-%d'))
        ax.xaxis.set_major_locator(mdates.WeekdayLocator())
        
        # 添加网格
        ax.grid(True, axis='x', alpha=0.3)
        
        # 添加图例
        ax.legend()
        
        plt.tight_layout()
        plt.show()
    
    def calculate_critical_path(self, tasks):
        """计算关键路径"""
        # 简化的关键路径计算
        # 实际应用中需要更复杂的算法
        total_duration = sum(task['duration'] for task in tasks)
        
        # 关键路径任务（假设所有任务都关键）
        critical_tasks = [task['name'] for task in tasks]
        
        return {
            'total_duration': total_duration,
            'critical_tasks': critical_tasks,
            'slack_times': {task['name']: 0 for task in tasks}
        }
    
    def create_project_dashboard(self, tasks, progress_data):
        """创建项目仪表板"""
        fig, axes = plt.subplots(2, 2, figsize=(15, 10))
        
        # 任务进度
        task_names = [task['name'] for task in tasks]
        progress = [progress_data.get(name, 0) for name in task_names]
        
        axes[0, 0].barh(task_names, progress)
        axes[0, 0].set_xlabel('完成进度 (%)')
        axes[0, 0].set_title('任务进度')
        axes[0, 0].set_xlim(0, 100)
        
        # 时间进度
        start = datetime.strptime("2023-01-01", "%Y-%m-%d")
        current_time = datetime.now()
        
        # 计算总天数
        total_days = 365
        elapsed_days = (current_time - start).days
        
        # 计算进度
        project_progress = min(elapsed_days / total_days * 100, 100)
        
        axes[0, 1].bar(['项目'], [project_progress], color='skyblue')
        axes[0, 1].set_ylim(0, 100)
        axes[0, 1].set_ylabel('进度 (%)')
        axes[0, 1].set_title('项目整体进度')
        
        # 风险分布
        risks = ['高', '中', '低']
        risk_counts = [2, 3, 5]
        
        axes[1, 0].bar(risks, risk_counts, color=['red', 'orange', 'green'])
        axes[1, 0].set_ylabel('风险数量')
        axes[1, 0].set_title('风险分布')
        
        # 资源使用
        resources = ['人力', '设备', '经费']
        resource_usage = [85, 70, 90]

        axes[1, 1].set_ylabel('使用率 (%)')
        axes[1, 1].set_title('资源使用情况')
        
        plt.tight_layout()
        plt.show()

class ResearchProject:
    """研究项目"""
    
    def __init__(self, title, start_date, end_date):
        """初始化"""
        self.title = title
        self.start_date = start_date
        self.end_date = end_date
        self.tasks = []
        self.team = []
        self.budget = 0
        self.status = "未开始"
    
    def add_task(self, name, duration, start_day, dependencies=None):
        """添加任务"""
        task = {
            'name': name,
            'duration': duration,
            'start': start_day,
            'dependencies': dependencies or [],
            'status': 'pending',
            'progress': 0
        }
        self.tasks.append(task)
        return task
    
    def add_team_member(self, name, role, expertise):
        """添加团队成员"""
        member = {
            'name': name,
            'role': role,
            'expertise': expertise
        }
        self.team.append(member)
        return member
    
    def set_budget(self, budget):
        """设置预算"""
        self.budget = budget
    
    def update_task_progress(self, task_name, progress):
        """更新任务进度"""
        for task in self.tasks:
            if task['name'] == task_name:
                task['progress'] = progress
                break
    
    def get_project_status(self):
        """获取项目状态"""
        # 计算平均进度
        if self.tasks:
            avg_progress = sum(task['progress'] for task in self.tasks) / len(self.tasks)
        else:
            avg_progress = 0
        
        # 确定项目状态
        if avg_progress < 30:
            status = "初期"
        elif avg_progress < 70:
            status = "中期"
        else:
            status = "后期"
        
        return {
            'title': self.title,
            'progress': avg_progress,
            'status': status,
            'tasks_count': len(self.tasks),
            'team_size': len(self.team),
            'budget': self.budget
        }

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from datetime import datetime, timedelta
import json

class QualityManagement:
    """质量管理"""
    
    def __init__(self):
        """初始化"""
        self.quality_standards = {}
        self.quality_metrics = {}
        self.quality_reports = []
    
    def set_quality_standard(self, process, standard):
        """设置质量标准"""
        self.quality_standards[process] = standard
    
    def calculate_quality_metric(self, actual, standard):
        """计算质量指标"""
        # 计算偏差
        deviation = actual - standard
        
        # 计算偏差率
        deviation_rate = abs(deviation) / standard * 100
        
        # 计算质量等级
        if deviation_rate <= 5:
            quality_level = "优秀"
        elif deviation_rate <= 10:
            quality_level = "良好"
        elif deviation_rate <= 20:
            quality_level = "合格"
        else:
            quality_level = "不合格"
        
        return {
            'deviation': deviation,
            'deviation_rate': deviation_rate,
            'quality_level': quality_level
        }
    
    def create_quality_control_chart(self, data, process_name):
        """创建质量控制图"""
        fig, ax = plt.subplots(figsize=(10, 6))
        
        # 计算控制限
        mean = np.mean(data)
        std = np.std(data)
        upper_control_limit = mean + 3 * std
        lower_control_limit = mean - 3 * std
        
        # 绘制数据点
        ax.plot(data, 'b-', linewidth=1, label='数据')
        
        # 绘制控制限
        ax.axhline(mean, color='r', linestyle='--', label='中心线')
        ax.axhline(upper_control_limit, color='g', linestyle='--', label='上控制限')
        ax.axhline(lower_control_limit, color='g', linestyle='--', label='下控制限')
        
        # 设置标题和标签
        ax.set_title(f'{process_name}质量控制图')
        ax.set_xlabel('样本序号')
        ax.set_ylabel('质量指标')
        ax.legend()
        ax.grid(True, alpha=0.3)
        
        plt.tight_layout()
        plt.show()
    
    def generate_quality_report(self, process_name, metrics):
        """生成质量报告"""
        report = {
            'process': process_name,
            'report_date': datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            'metrics': metrics,
            'recommendations': []
        }
        
        # 生成建议
        for metric_name, metric_value in metrics.items():
            if metric_value['quality_level'] == '不合格':
                report['recommendations'].append(f"需要改进{metric_name}质量")
            elif metric_value['quality_level'] == '良好':
                report['recommendations'].append(f"保持{metric_name}质量")
        
        self.quality_reports.append(report)
        return report

class RiskManagement:
    """风险管理"""
    
    def __init__(self):
        """初始化"""
        self.risks = []
        self.risk_matrix = {}
    
    def add_risk(self, name, probability, impact, category):
        """添加风险"""
        risk = {
            'name': name,
            'probability': probability,
            'impact': impact,
            'category': category,
            'risk_score': probability * impact,
            'status': '识别',
            'response': '未制定'
        }
        self.risks.append(risk)
        return risk
    
    def calculate_risk_matrix(self):
        """计算风险矩阵"""
        matrix = {
            'high_probability_high_impact': [],
            'high_probability_low_impact': [],
            'low_probability_high_impact': [],
            'low_probability_low_impact': []
        }
        
        for risk in self.risks:
            if risk['probability'] >= 0.7 and risk['impact'] >= 0.7:
                matrix['high_probability_high_impact'].append(risk)
            elif risk['probability'] >= 0.7:
                matrix['high_probability_low_impact'].append(risk)
            elif risk['impact'] >= 0.7:
                matrix['low_probability_high_impact'].append(risk)
            else:
                matrix['low_probability_low_impact'].append(risk)
        
        self.risk_matrix = matrix
        return matrix
    
    def create_risk_dashboard(self):
        """创建风险仪表板"""
        fig, axes = plt.subplots(2, 2, figsize=(15, 10))
        
        # 风险分布
        categories = [risk['category'] for risk in self.risks]
        category_counts = pd.Series(categories).value_counts()
        
        axes[0, 0].pie(category_counts.values, labels=category_counts.index, autopct='%1.1f%%')
        axes[0, 0].set_title('风险分布')
        
        # 风险评分分布
        risk_scores = [risk['risk_score'] for risk in self.risks]
        axes[0, 1].hist(risk_scores, bins=10, alpha=0.7, color='skyblue')
        axes[0, 1].set_xlabel('风险评分')
        axes[0, 1].set_ylabel('风险数量')
        axes[0, 1].set_title('风险评分分布')
        axes[0, 1].grid(True, alpha=0.3)
        
        # 风险优先级
        sorted_risks = sorted(self.risks, key=lambda x: x['risk_score'], reverse=True)
        risk_names = [risk['name'] for risk in sorted_risks[:5]]
        risk_scores = [risk['risk_score'] for risk in sorted_risks[:5]]
        
        axes[1, 0].barh(risk_names, risk_scores, color='salmon')
        axes[1, 0].set_xlabel('风险评分')
        axes[1, 0].set_title('风险优先级（前5）')
        
        # 风险应对状态
        response_status = [risk['response'] for risk in self.risks]
        response_counts = pd.Series(response_status).value_counts()
        
        axes[1, 1].bar(response_counts.index, response_counts.values, color='lightgreen')
        axes[1, 1].set_xlabel('应对状态')
        axes[1, 1].set_ylabel('风险数量')
        axes[1, 1].set_title('风险应对状态')
        
        plt.tight_layout()
        plt.show()
    
    def generate_risk_report(self):
        """生成风险报告"""
        matrix = self.calculate_risk_matrix()
        
        report = {
            'report_date': datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            'total_risks': len(self.risks),
            'risk_matrix': matrix,
            'high_risk_count': len(matrix['high_probability_high_impact']),
            'recommendations': []
        }
        
        # 生成建议
        if report['high_risk_count'] > 0:

class ResearchProjectManagement:
    """研究项目管理"""
    
    def __init__(self, title):
        """初始化"""
        self.title = title
        self.start_date = None
        self.end_date = None
        self.status = "未开始"
        self.tasks = []
        self.team = []
        self.budget = 0
        self.quality_manager = QualityManagement()
        self.risk_manager = RiskManagement()
    
    def set_project_dates(self, start_date, end_date):
        """设置项目日期"""
        self.start_date = start_date
        self.end_date = end_date
    
    def add_task(self, name, duration, start_day, dependencies=None):
        """添加任务"""
        task = {
            'name': name,
            'duration': duration,
            'start': start_day,
            'dependencies': dependencies or [],
            'status': 'pending',
            'progress': 0,
            'actual_start': None,
            'actual_end': None
        }
        self.tasks.append(task)
        return task
    
    def add_team_member(self, name, role, expertise):
        """添加团队成员"""
        member = {
            'name': name,
            'role': role,
            'expertise': expertise,
            'tasks': []
        }
        self.team.append(member)
        return member
    
    def add_risk(self, name, probability, impact, category):
        """添加风险"""
        return self.risk_manager.add_risk(name, probability, impact, category)
    
    def set_budget(self, budget):
        """设置预算"""
        self.budget = budget
    
    def update_task_progress(self, task_name, progress, actual_start=None, actual_end=None):
        """更新任务进度"""
        for task in self.tasks:
            if task['name'] == task_name:
                task['progress'] = progress
                if actual_start:
                    task['actual_start'] = actual_start
                if actual_end:
                    task['actual_end'] = actual_end
                break
    
    def get_project_status(self):
        """获取项目状态"""
        # 计算平均进度
        if self.tasks:
            avg_progress = sum(task['progress'] for task in self.tasks) / len(self.tasks)
        else:
            avg_progress = 0
        
        # 计算项目周期
        if self.start_date and self.end_date:
            total_days = (self.end_date - self.start_date).days
            elapsed_days = (datetime.now() - self.start_date).days
            progress_percentage = min(elapsed_days / total_days * 100, 100)
        else:
            total_days = 0
            elapsed_days = 0
            progress_percentage = 0
        
        # 确定项目状态
        if avg_progress < 30:
            status = "初期"
        elif avg_progress < 70:
            status = "中期"
        else:
            status = "后期"
        
        return {
            'title': self.title,
            'status': status,
            'progress': avg_progress,
            'progress_percentage': progress_percentage,
            'total_days': total_days,
            'elapsed_days': elapsed_days,
            'tasks_count': len(self.tasks),
            'team_size': len(self.team),
            'budget': self.budget
        }
    
    def create_project_dashboard(self):
        """创建项目仪表板"""
        fig, axes = plt.subplots(2, 2, figsize=(15, 10))
        
        # 项目进度
        status = self.get_project_status()
        axes[0, 0].bar(['项目'], [status['progress']], color='skyblue')
        axes[0, 0].set_ylim(0, 100)
        axes[0, 0].set_ylabel('进度 (%)')
        axes[0, 0].set_title('项目整体进度')
        
        # 任务分布
        task_names = [task['name'] for task in self.tasks]
        task_progress = [task['progress'] for task in self.tasks]
        
        axes[0, 1].barh(task_names, task_progress, color='lightgreen')
        axes[0, 1].set_xlabel('完成进度 (%)')
        axes[0, 1].set_title('任务进度')
        axes[0, 1].set_xlim(0, 100)
        
        # 团队规模
        axes[1, 0].bar(['团队'], [status['team_size']], color='salmon')
        axes[1, 0].set_ylabel('人数')
        axes[1, 0].set_title('团队规模')
        
        # 预算使用
        budget_usage = min(status['budget'] / 100000 * 100, 100) if status['budget'] > 0 else 0
        axes[1, 1].bar(['预算'], [budget_usage], color='gold')
        axes[1, 1].set_ylim(0, 100)
        axes[1, 1].set_ylabel('使用率 (%)')
        axes[1, 1].set_title('预算使用情况')
        
        plt.tight_layout()
        plt.show()
    
    def generate_project_report(self):
        """生成项目报告"""
        status = self.get_project_status()
        quality_report = self.quality_manager.generate_quality_report("项目质量", {})
        risk_report = self.risk_manager.generate_risk_report()
        
        report = {
            'project': self.title,
            'report_date': datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            'status': status,
            'quality_report': quality_report,
            'risk_report': risk_report,
            'recommendations': []
        }
        
        # 生成建议
        if status['progress'] < 50:
            report['recommendations'].append("加快项目进度")
        if risk_report['high_risk_count'] > 0:
            report['recommendations'].append("优先处理高风险项目")
        
        return report
