# 多目标优化实现

## 项目概述
- **目标**: 构建多目标投资组合优化系统，同时考虑收益、风险、流动性等多个目标
- **时间**: 2025年11月 - 2026年2月
- **技术栈**: Python (scipy, cvxpy, pandas), MOSEK求解器
- **核心成果**: 实现了均值-方差、风险平价、Black-Litterman等多种优化模型

## 项目背景与动机

单个Alpha策略虽然能产生收益，但如何将这些收益转化为实际可投资的组合？组合优化要解决的核心问题是：**在给定约束下，如何最优地分配资金？**

这个项目源于以下思考：
1. **收益与风险的权衡**: 单纯追求高收益会导致风险过大
2. **分散化的数学表达**: 如何量化"不把鸡蛋放一个篮子"
3. **主观观点的融合**: 如何将分析师的观点融入组合

## 技术实现

### 核心优化模型

#### 1. 均值-方差优化 (Mean-Variance Optimization)
```python
import cvxpy as cp
import numpy as np
import pandas as pd

class MeanVarianceOptimizer:
    """均值-方差优化器"""
    
    def __init__(self, expected_returns, cov_matrix, risk_aversion=1.0):
        self.expected_returns = expected_returns
        self.cov_matrix = cov_matrix
        self.risk_aversion = risk_aversion
        self.n_assets = len(expected_returns)
    
    def optimize(self, target_return=None):
        """优化组合权重"""
        w = cp.Variable(self.n_assets)
        
        # 目标函数：最大化收益 - 风险惩罚
        portfolio_return = self.expected_returns @ w
        portfolio_variance = cp.quad_form(w, self.cov_matrix)
        
        objective = portfolio_return - self.risk_aversion * portfolio_variance
        
        constraints = [
            cp.sum(w) == 1,           # 权重和为1
            w >= 0,                    # 不允许做空
            w <= 0.15                  # 单个资产最大权重15%
        ]
        
        # 如果设置了目标收益约束
        if target_return is not None:
            constraints.append(portfolio_return >= target_return)
        
        problem = cp.Problem(cp.Maximize(objective), constraints)
        problem.solve(solver=cp.ECOS)
        
        return {
            'weights': w.value,
            'expected_return': portfolio_return.value,
            'expected_volatility': cp.sqrt(portfolio_variance).value
        }
```

#### 2. 风险平价 (Risk Parity)
```python
class RiskParityOptimizer:
    """风险平价优化器"""
    
    def __init__(self, cov_matrix):
        self.cov_matrix = cov_matrix
        self.n_assets = cov_matrix.shape[0]
    
    def _risk_contribution(self, w, cov_matrix):
        """计算各资产的风险贡献"""
        portfolio_vol = np.sqrt(w @ cov_matrix @ w)
        marginal_risk = cov_matrix @ w
        risk_contrib = w * marginal_risk / portfolio_vol
        return risk_contrib
    
    def optimize(self, target_vol=None):
        """优化使各资产风险贡献相等"""
        w = cp.Variable(self.n_assets)
        
        # 风险贡献
        portfolio_vol = cp.sqrt(cp.quad_form(w, self.cov_matrix))
        marginal_risk = self.cov_matrix @ w
        risk_contrib = w * marginal_risk / portfolio_vol
        
        # 目标：使各资产风险贡献相等
        target_risk_contrib = portfolio_vol / self.n_assets
        
        # 最小化与目标风险贡献的偏差
        risk_diff = risk_contrib - target_risk_contrib
        
        objective = cp.sum_squares(risk_diff)
        
        constraints = [
            cp.sum(w) == 1,
            w >= 0.01,  # 最小权重1%
            w <= 0.30
        ]
        
        if target_vol is not None:
            constraints.append(portfolio_vol == target_vol)
        
        problem = cp.Problem(cp.Minimize(objective), constraints)
        problem.solve(solver=cp.SCS)
        
        return {
            'weights': w.value,
            'risk_contributions': self._risk_contribution(w.value, self.cov_matrix)
        }
```

#### 3. Black-Litterman模型
```python
class BlackLittermanOptimizer:
    """Black-Litterman模型：融合市场均衡与主观观点"""
    
    def __init__(self, market_caps, cov_matrix, risk_aversion=2.5):
        self.market_caps = market_caps
        self.cov_matrix = cov_matrix
        self.risk_aversion = risk_aversion
        self.n_assets = len(market_caps)
        
        # 计算市场均衡收益
        self.market_weights = market_caps / market_caps.sum()
        self.pi = risk_aversion * cov_matrix @ self.market_weights
    
    def blend_views(self, views, view_confidences):
        """
        融合主观观点
        views: {asset: expected_return}
        view_confidences: [0, 1] 信心水平
        """
        # 构建观点矩阵P和Q
        assets = list(views.keys())
        n_views = len(views)
        
        P = np.zeros((n_views, self.n_assets))
        Q = np.zeros((n_views, 1))
        Omega = np.zeros((n_views, n_views))  # 观点不确定性
        
        for i, (asset, view_return) in enumerate(views.items()):
            asset_idx = list(self.market_caps.index).index(asset)
            P[i, asset_idx] = 1
            Q[i] = view_return
            Omega[i, i] = (1 - view_confidences[i]) * self.cov_matrix[asset_idx, asset_idx]
        
        # Black-Litterman公式
        MLE_returns = self._compute_bl_returns(P, Q, Omega)
        
        return MLE_returns
    
    def _compute_bl_returns(self, P, Q, Omega):
        """计算融合后的收益"""
        tau = 0.1  # 缩放因子
        
        MLE = self.pi + tau * P.T @ np.linalg.inv(tau * P @ self.cov_matrix @ P.T + Omega) @ (Q - P @ self.pi)
        
        return pd.Series(MLE.flatten(), index=self.market_caps.index)
    
    def optimize(self, expected_returns=None):
        """使用BL收益进行优化"""
        if expected_returns is None:
            expected_returns = self.pi
        
        optimizer = MeanVarianceOptimizer(expected_returns, self.cov_matrix)
        return optimizer.optimize()
```

## 多目标优化框架

```python
class MultiObjectiveOptimizer:
    """多目标优化：同时考虑多个目标"""
    
    def __init__(self, returns, cov_matrix, volumes):
        self.returns = returns
        self.cov_matrix = cov_matrix
        self.volumes = volumes
        self.n_assets = len(returns)
    
    def optimize_pareto(self, n_points=20):
        """计算Pareto前沿"""
        pareto_weights = []
        
        # 在不同风险水平下优化
        min_return = self.returns.min()
        max_return = self.returns.max()
        return_range = np.linspace(min_return, max_return, n_points)
        
        for target_return in return_range:
            w = cp.Variable(self.n_assets)
            
            portfolio_return = self.returns @ w
            portfolio_vol = cp.sqrt(cp.quad_form(w, self.cov_matrix))
            
            # 最小化换手率（近似）
            turnover = cp.norm(w - self.market_weights, 1)
            
            objective = portfolio_vol + 0.1 * turnover
            
            constraints = [
                cp.sum(w) == 1,
                w >= 0,
                w <= 0.20,
                portfolio_return >= target_return
            ]
            
            try:
                problem = cp.Problem(cp.Minimize(objective), constraints)
                problem.solve(solver=cp.SCS, verbose=False)
                
                if problem.status == 'optimal':
                    pareto_weights.append({
                        'weights': w.value,
                        'return': target_return,
                        'volatility': portfolio_vol.value
                    })
            except:
                continue
        
        return pareto_weights
```

## 挑战与解决方案

**挑战: 凸优化问题的数值稳定性**
- 某些参数组合会导致求解器收敛困难
- 通过参数缩放和约束松弛解决

**挑战: 过度集中于少数资产**
- 标准优化往往给出极端权重
- 通过增加权重约束和正则化解决
