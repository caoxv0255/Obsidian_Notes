import jqdatasdk as jq
from jqdatasdk import auth, get_price, get_fundamentals

class JoinQuantBroker:
    """聚宽平台对接"""

            pos['code']: {
                'shares': pos['shares'],

                'current_price': pos['price']

        }

        """获取账户信息"""
        account_info = jq.get_account()
        return {
            'cash': account_info['cash'],
            'market_value': account_info['market_value'],
            'total_value': account_info['total_value'],
            'today_profit': account_info['today_profit']
        }
    
    def place_order(self, code, shares, action):
        """下单
        action: 'buy' or 'sell'
        """
        if action == 'buy':
            return jq.order(code, shares, 'buy')
        else:
            return jq.order(code, shares, 'sell')
    
    def cancel_order(self, order_id):
        """撤单"""
        return jq.cancel_order(order_id)

## 本地模拟交易引擎

对于不想用真实API的场景，本地模拟引擎提供了完整的交易模拟：

```python
class LocalSimulator:
    """本地模拟交易引擎"""
    
    def __init__(self, initial_capital=1000000):
        self.initial_capital = initial_capital
        self.cash = initial_capital
        self.positions = {}  # {code: {'shares': int, 'avg_cost': float}}
        self.orders = []
        self.trades = []
        self.daily_value = []
    
    def receive_signal(self, signal_date, signals):
        """接收策略信号
        signals: {code: ('buy'/'sell', shares_or_pct)}
        """
        for code, (action, amount) in signals.items():
            if action == 'buy':
                self._execute_buy(signal_date, code, amount)
            elif action == 'sell':
                self._execute_sell(signal_date, code, amount)
    
    def _execute_buy(self, date, code, amount):
        """执行买入"""
        # 获取当日价格（模拟使用收盘价）
        price = self._get_price(code, date)
        
        # 计算买入数量（整手）
        shares = int(amount / price / 100) * 100
        
        if shares > 0:
            cost = shares * price * 1.0003  # 包含手续费
            
            if cost <= self.cash:
                self.orders.append({
                    'date': date,
                    'code': code,
                    'action': 'buy',
                    'shares': shares,
                    'price': price
                })
                
                # 更新持仓
                if code in self.positions:
                    old_shares = self.positions[code]['shares']
                    old_cost = self.positions[code]['avg_cost']
                    new_shares = old_shares + shares
                    new_cost = (old_shares * old_cost + shares * price) / new_shares
                    self.positions[code] = {
                        'shares': new_shares,
                        'avg_cost': new_cost
                    }
                else:
                    self.positions[code] = {
                        'shares': shares,
                        'avg_cost': price
                    }
                
                self.cash -= cost
                
                self.trades.append({
                    'date': date,
                    'code': code,
                    'action': 'buy',
                    'shares': shares,
                    'price': price,
                    'cost': cost
                })
    
    def _execute_sell(self, date, code, amount):
        """执行卖出"""
        if code not in self.positions:
            return
        
        price = self._get_price(code, date)
        current_shares = self.positions[code]['shares']
        
        # 全部卖出或按数量卖出
        shares = min(current_shares, int(amount / 100) * 100) if isinstance(amount, (int, float)) else current_shares
        
        if shares > 0:
            proceeds = shares * price * 0.9997  # 扣除手续费
            
            self.orders.append({
                'date': date,
                'code': code,
                'action': 'sell',
                'shares': shares,
                'price': price
            })
            
            self.positions[code]['shares'] -= shares
            if self.positions[code]['shares'] == 0:
                del self.positions[code]
            
            self.cash += proceeds
            
            self.trades.append({
                'date': date,
                'code': code,
                'action': 'sell',
                'shares': shares,
                'price': price,
                'proceeds': proceeds
            })
    
    def get_portfolio_value(self, date):
        """计算组合市值"""
        position_value = 0
        for code, pos in self.positions.items():
            price = self._get_price(code, date)
            position_value += pos['shares'] * price
        
        return self.cash + position_value
    
    def get_daily_report(self, date):
        """生成每日报告"""
        portfolio_value = self.get_portfolio_value(date)
        
        return {
            'date': date,
            'cash': self.cash,
            'position_value': portfolio_value - self.cash,
            'total_value': portfolio_value,
            'positions': self.positions.copy(),
            'orders_count': len([o for o in self.orders if o['date'] == date])
        }
```

### 仓位管理
```python
class PositionManager:
    """仓位管理器"""
    
    def __init__(self, max_single_position=0.15, max_sector=0.3):
        self.max_single_position = max_single_position    # 单股最大仓位15%
        self.max_sector = max_sector                      # 行业最大仓位30%
    
    def validate_signal(self, signal, current_positions, portfolio_value, sector_map):
        """验证信号是否合规"""
        code = signal['code']
        action = signal['action']
        shares = signal['shares']
        
        # 单股仓位检查
        if action == 'buy':
            new_position_value = shares * signal['price']
            position_pct = new_position_value / portfolio_value
            
            if position_pct > self.max_single_position:
                # 缩减仓位到上限
                max_shares = int(portfolio_value * self.max_single_position / signal['price'] / 100) * 100
                signal['shares'] = max_shares
        
        # 行业仓位检查
        if code in sector_map:
            sector = sector_map[code]
            current_sector_value = sum(
                pos['shares'] * pos.get('price', pos['avg_cost'])
                for code_, pos in current_positions.items()
                if sector_map.get(code_) == sector
            )
            
            # 如果是新买入，加上拟买入的市值
            if action == 'buy':
                new_sector_value = current_sector_value + signal['shares'] * signal['price']
            else:
                new_sector_value = current_sector_value
            
            if new_sector_value / portfolio_value > self.max_sector:
                return False  # 行业仓位超限，拒绝交易
        
        return True
```

## 模拟vs实盘的关键差异

| 维度 | 模拟交易 | 实盘 |
|------|----------|------|
| 执行价格 | 收盘价/固定滑点 | 实时波动 |
| 成交确认 | 即时 | 可能有延迟 |
| 仓位限制 | 按比例 | 按资金实际可用 |
| 涨跌停 | 可交易 | 买不到/卖不出 |
| 流动性 | 假设充足 | 可能不足 |

**核心认知**: 模拟交易永远无法完全复制实盘体验，但它是低成本验证策略逻辑的最佳方式。
