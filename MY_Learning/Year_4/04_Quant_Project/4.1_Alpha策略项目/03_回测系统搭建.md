# 回测系统搭建

## 回测系统架构

一个可靠的回测系统需要解决几个核心问题：避免未来函数、考虑交易成本、模拟真实交易执行。本项目搭建了模块化的回测框架。

```python
class DataLoader:
    """统一数据加载器"""
    
    def __init__(self, data_dir):
        self.data_dir = data_dir
    
    def load_daily_data(self, start_date, end_date):
        """加载日频行情数据"""
        price = pd.read_parquet(f'{self.data_dir}/price.parquet')
        price = price[(price['date'] >= start_date) & (price['date'] <= end_date)]
        return price
    
    def load_fundamental(self, dates):
        """加载财务数据"""
        fund = pd.read_parquet(f'{self.data_dir}/fundamental.parquet')
        # 财务数据对齐：使用最新可用数据（避免未来函数）
        fund = fund[fund['announce_date'] <= dates]
        fund = fund.sort_values('announce_date').groupby(['code', 'report_date']).last()
        return fund
```

### 数据预处理：避免常见陷阱
```python
class DataPreprocessor:
    """数据预处理：处理停牌、涨跌停、退市等"""
    
    def handle_suspended(self, price_df):
        """处理停牌：停牌期间保持前一交易日收盘价"""
        return price_df.fillna(method='ffill')
    
    def handle_limit_up_down(self, price_df, returns_df):
        """处理涨跌停：涨跌停日不可交易"""
        limit_up = returns_df > 0.095  # 涨跌停阈值
        limit_down = returns_df < -0.095
        # 将涨跌停日的收益设为0（无法买入/卖出）
        returns_adj = returns_df.mask(limit_up | limit_down, 0)
        return returns_adj
    
    def remove_delisted(self, price_df, delist_date):
        """剔除退市股票"""
        return price_df[price_df['date'] < delist_date]
```

## 回测引擎核心

### 事件驱动回测
```python
class BacktestEngine:
    """事件驱动回测引擎"""
    
    def __init__(self, initial_capital, commission_rate, slippage_rate):
        self.initial_capital = initial_capital
        self.commission_rate = commission_rate
        self.slippage_rate = slippage_rate
        
        # 状态变量
        self.cash = initial_capital
        self.positions = {}  # {stock_code: shares}
        self.portfolio_value = initial_capital
        self.trade_log = []
        self.daily_stats = []
    
    def run(self, strategy, price_data, factor_data, dates):
        """运行回测"""
        for date in dates:
            # 1. 获取当日数据
            prices = price_data.xs(date, level='date')
            factors = factor_data.xs(date, level='date') if factor_data is not None else None
            
            # 2. 生成信号
            signals = strategy.generate_signals(factors, date)
            
            # 3. 计算当前持仓价值
            position_value = sum(
                self.positions.get(code, 0) * prices.loc[code, 'close']
                for code in self.positions
            )
            
            # 4. 交易执行
            self.execute_trades(signals, prices, date)
            
            # 5. 更新组合价值
            self.update_portfolio_value(prices, date)
            
            # 6. 记录统计
            self.record_daily_stats(date)
    
    def execute_trades(self, signals, prices, date):
        """执行交易：考虑交易成本和滑点"""
        for code, signal in signals.items():
            if code not in prices.index:
                continue
                
            current_position = self.positions.get(code, 0)
            target_position = signal  # 假设signal表示目标持仓方向（1, 0, -1）
            
            if target_position != current_position:
                # 计算交易数量
                trade_value = abs(target_position - current_position) * self.portfolio_value * 0.1
                shares = int(trade_value / prices.loc[code, 'close'] / 100) * 100  # 整手
                
                if shares > 0:
                    # 买入
                    if target_position > current_position:
                        cost = shares * prices.loc[code, 'close']
                        commission = cost * self.commission_rate
                        slippage = cost * self.slippage_rate
                        total_cost = cost + commission + slippage
                        
                        self.cash -= total_cost
                        self.positions[code] = self.positions.get(code, 0) + shares
                        
                    # 卖出
                    else:
                        proceeds = shares * prices.loc[code, 'close']
                        commission = proceeds * self.commission_rate
                        slippage = proceeds * self.slippage_rate
                        net_proceeds = proceeds - commission - slippage
                        
                        self.cash += net_proceeds
                        self.positions[code] = self.positions.get(code, 0) - shares
                        
                        # 清空仓位
                        if self.positions[code] == 0:
                            del self.positions[code]
                    
                    # 记录交易
                    self.trade_log.append({
                        'date': date,
                        'code': code,
                        'action': 'buy' if target_position > current_position else 'sell',
                        'shares': shares,
                        'price': prices.loc[code, 'close'],
                        'commission': commission,
                        'slippage': slippage
                    })
    
    def update_portfolio_value(self, prices, date):
        """更新组合市值"""
        position_value = sum(
            self.positions.get(code, 0) * prices.loc[code, 'close']
            for code in self.positions
            if code in prices.index
        )
        self.portfolio_value = self.cash + position_value
```

## 绩效分析

### 核心指标计算
```python
class PerformanceAnalyzer:
    """绩效分析器"""
    
    def calculate_returns(self, portfolio_value_series):
        """计算收益率序列"""
        returns = portfolio_value_series.pct_change().dropna()
        return returns
    
    def annual_return(self, returns):
        """年化收益率"""
        n_periods = len(returns)
        n_years = n_periods / 252
        cumulative_return = (1 + returns).prod() - 1
        return (1 + cumulative_return) ** (1 / n_years) - 1
    
    def sharpe_ratio(self, returns, risk_free_rate=0.03):
        """夏普比率"""
        excess_returns = returns - risk_free_rate / 252
        return np.sqrt(252) * excess_returns.mean() / excess_returns.std()
    
    def max_drawdown(self, portfolio_value_series):
        """最大回撤"""
        cummax = portfolio_value_series.cummax()
        drawdown = (portfolio_value_series - cummax) / cummax
        return drawdown.min()
    
    def calmar_ratio(self, returns, portfolio_value_series):
        """Calmar比率：年化收益/最大回撤"""
        ann_return = self.annual_return(returns)
        max_dd = abs(self.max_drawdown(portfolio_value_series))
        return ann_return / max_dd if max_dd != 0 else 0
    
    def win_rate(self, returns):
        """胜率"""
        return (returns > 0).sum() / len(returns)
    
    def profit_loss_ratio(self, returns):
        """盈亏比"""
        avg_win = returns[returns > 0].mean()
        avg_loss = abs(returns[returns < 0].mean())
        return avg_win / avg_loss if avg_loss != 0 else 0
```

## 回测常见陷阱与规避

### 1. 未来函数 (Look-ahead Bias)
**问题**: 使用了当期才能获得的数据
**解决**: 
- 财务数据使用实际公告日期
- 市值数据使用开盘价或前一天收盘价

### 2. 存活者偏差 (Survivorship Bias)
**问题**: 只用当前存活的股票回测
**解决**: 购买包含退市股票的历史数据库

### 3. 前视偏差 (Foresight Bias)
**问题**: 信号生成和执行在同一天
**解决**: 假设信号在收盘后生成，次日开盘执行

### 4. 过度拟合 (Overfitting)
**问题**: 策略参数在历史数据上过优化
**解决**:
- 样本外测试（out-of-sample）
- Walk-forward分析
- 参数稳健性检验

### 5. 流动性忽视
**问题**: 假设可以买卖任意数量
**解决**:
- 设置单笔最大交易量
- 加入滑点模型
- 考虑冲击成本
