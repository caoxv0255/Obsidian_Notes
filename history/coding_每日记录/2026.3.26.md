### 学习内容总结

| 维度 | 邻接矩阵 | 邻接表 | 适用场景 |
|------|---------|--------|---------|

| 查询边 | O(1) | O(deg(u)) | 邻接矩阵查边快 |
| 遍历邻居 | O(n) | O(deg(u)) | 邻接表遍历快 |
| 动态增顶点 | 困难 | 简单 | 动态图选邻接表 |

## 今日LeetCode题目

### LeetCode 1971 - 寻找图中是否存在路径

**题目描述：**
- 无向图，n个顶点，edges数组表示边
- 判断从source到destination是否存在有效路径

**三种解法实现：**

#### 方法1：BFS（广度优先搜索）

```cpp
class Solution {
public:
    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        if (source == destination) return true;
        
        // 构建邻接表
        vector<vector<int>> adjList(n);
        for (auto& edge : edges) {
            int u = edge[0], v = edge[1];
            adjList[u].push_back(v);
            adjList[v].push_back(u);
        }
        
        // BFS遍历
        vector<bool> visited(n, false);
        queue<int> q;
        q.push(source);
        visited[source] = true;
        
        while (!q.empty()) {
            int node = q.front();
            q.pop();
            
            for (int neighbor : adjList[node]) {
                if (neighbor == destination) return true;
                if (!visited[neighbor]) {
                    visited[neighbor] = true;
                    q.push(neighbor);
                }
            }
        }
        
        return false;
    }
};
```

**复杂度：** 时间 O(n+m)，空间 O(n+m)

#### 方法2：DFS（深度优先搜索）

```cpp
class Solution {
public:
    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        if (source == destination) return true;
        
        vector<vector<int>> adjList(n);
        for (auto& edge : edges) {
            int u = edge[0], v = edge[1];
            adjList[u].push_back(v);
            adjList[v].push_back(u);
        }
        
        vector<bool> visited(n, false);
        return dfs(source, destination, adjList, visited);
    }
    
private:
    bool dfs(int node, int dest, vector<vector<int>>& adjList, vector<bool>& visited) {
        if (node == dest) return true;
        visited[node] = true;
        
        for (int neighbor : adjList[node]) {
            if (!visited[neighbor]) {
                if (dfs(neighbor, dest, adjList, visited)) {
                    return true;
                }
            }
        }
        return false;
    }
};
```

#### 方法3：并查集（Union-Find）

```cpp
class UnionFind {
public:
    UnionFind(int n) : parent(n), rank(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }
    
    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    
    void unite(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        
        if (rootX != rootY) {
            if (rank[rootX] < rank[rootY]) {
                parent[rootX] = rootY;
            } else if (rank[rootX] > rank[rootY]) {
                parent[rootY] = rootX;
            } else {
                parent[rootY] = rootX;
                rank[rootX]++;
            }
        }
    }
    
    bool connected(int x, int y) {
        return find(x) == find(y);
    }
    
private:
    vector<int> parent;
    vector<int> rank;
};

class Solution {
public:
    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        UnionFind uf(n);
        
        for (auto& edge : edges) {
            uf.unite(edge[0], edge[1]);
        }
        
        return uf.connected(source, destination);
    }
};
```

## 三种方法对比

| 方法 | 核心思想 | 优势 | 劣势 | 适用场景 |
|------|---------|------|------|---------|
| BFS | 逐层向外探索 | 天然找到最短路径，无递归栈溢出风险 | 需要队列空间 | 需要最短路径、层级遍历 |
| DFS | 一条路走到底 | 实现简洁，节省队列空间 | 递归深度可能栈溢出 | 需要路径回溯、拓扑排序 |
| 并查集 | 集合合并查询 | 查询连通性效率极高 | 不记录具体路径 | 仅需判断连通性 |

## 核心代码模板

**邻接表构建（无向图）：**
```cpp
vector<vector<int>> adjList(n);
for (auto& edge : edges) {
    int u = edge[0], v = edge[1];
    adjList[u].push_back(v);
    adjList[v].push_back(u);  // 无向图双向添加
}
```

**邻接表构建（有向图）：**
```cpp
vector<vector<int>> adjList(n);
for (auto& edge : edges) {
    int u = edge[0], v = edge[1];
    adjList[u].push_back(v);  // 有向图单向添加
}
```

**加权图邻接表：**
