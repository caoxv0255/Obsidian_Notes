### 重点知识点
- **链表节点结构**：`struct ListNode { int val; ListNode* next; }`
- **头插法反转**：保存next→断开curr→指向newHead→更新newHead→移动curr
- **时间复杂度**：链表遍历O(n)，反转链表O(n)，头部插入/删除O(1)
- **递归理解**：反转剩余部分→让下一个节点指向当前节点→当前节点指向null
- **链表与图的关系**：链表是有向线性图，是图算法的基础
- **栈的应用场景**：
  - 函数调用追踪（636题）：栈记录活跃函数，维护时间戳
  - 逆波兰表达式求值（150题）：栈存储操作数，遇运算符弹出计算
  - 模拟栈操作（1441题）：直接模拟Push/Pop操作构建结果

### 代码实现（C++头插法）
```cpp
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* newHead = nullptr;
        ListNode* curr = head;

        while (curr != nullptr) {
            ListNode* nextTemp = curr->next;
            curr->next = newHead;
            newHead = curr;
            curr = nextTemp;
        }

        return newHead;
    }
};
```

**LeetCode 636 - 函数的独占时间（栈+时间解析）**
```cpp
class Solution {
public:
    static inline int parseNum(const char* str, int len) {
        int result = 0;
        for (int i = 0; i < len; ++i) {
            result = result * 10 + (str[i] - '0');
        }
        return result;
    }

    static inline void parseLog(const string& log, int& id, bool& isStart,
                         int& timeStamp) {
        int pos = 0;
        int n = log.size();

        int idStart = pos;
        while (pos < n && log[pos] != ':') {
            pos++;
        }
        id = parseNum(log.data(), pos - idStart);
        pos++;

        isStart = (log[pos] == 's');
        while (pos < n && log[pos] != ':') {
            pos++;
        }
        pos++;

        timeStamp = parseNum(log.data() + pos, n - pos);
    }

    vector<int> exclusiveTime(int n, vector<string>& logs) {
        vector<int> res(n, 0);
        vector<int> stk;
        stk.reserve(logs.size());
        int prevTime = 0;

        for (const string& log : logs) {
            int id;
            bool isStart;
            int currTime;
            parseLog(log, id, isStart, currTime);

            if (isStart) {
                if (!stk.empty()) {
                    res[stk.back()] += currTime - prevTime;
                }
                stk.push_back(id);
                prevTime = currTime;
            } else {
                res[id] += currTime - prevTime + 1;
                stk.pop_back();
                prevTime = currTime + 1;
            }
        }

        return res;
    }
};
```

**LeetCode 150 - 逆波兰表达式求值（栈）**
```cpp
class Solution {
public:
    static inline int parseNum(const string& s) {
        int result = 0;
        int sign = 1;
        size_t i = 0;

        if (s[0] == '-') {
            sign = -1;
            i = 1;
        }

        for (; i < s.size(); ++i) {
            result = result * 10 + (s[i] - '0');
        }

        return result * sign;
    }

    int evalRPN(vector<string>& tokens) {
        vector<int> num;
        num.reserve((tokens.size() + 1) / 2);

        for (const string& token : tokens) {
            if (token.size() > 1 ||
                isdigit(static_cast<unsigned char>(token[0]))) {
                num.push_back(parseNum(token));
            } else {
                int b = num.back();
                num.pop_back();
                int a = num.back();
                num.pop_back();

                switch (token[0]) {
                case '+':
                    num.push_back(a + b);
                    break;
                case '-':
                    num.push_back(a - b);
                    break;
                case '*':
                    num.push_back(a * b);
                    break;
                case '/':
                    num.push_back(a / b);
                    break;
                }
            }
        }

        return num.back();
    }
};
```
