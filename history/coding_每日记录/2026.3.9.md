## 💡 学习与收获

### 核心收获（3个）
1. **单调栈的核心思想**：单调递增栈用于寻找"下一个更小/相等元素"，单调递减栈用于寻找"下一个更大/相等元素"。通过维护栈的单调性，可以在O(n)时间内解决需要"寻找最近满足条件元素"的问题
2. **从暴力到优化的思维转变**：柱状图最大矩形问题从暴力解法O(n²)优化到单调栈O(n)，关键在于利用单调栈一次性计算每个柱子的左右边界，避免了重复遍历
3. **栈的多种应用场景**：括号匹配（经典的栈应用）、每日温度（寻找下一个更高温度）、商品最终价格（寻找下一个折扣价），这些看似不同的问题本质上都可以用单调栈统一解决

### 疑问与思考（1个）
- 在单调栈问题中，为什么"从右向左遍历"和"从左向右遍历"都可以得到正确结果？理解了两种遍历方式的本质区别：从右向左遍历找右边第一个满足条件的元素，从左向右遍历找左边第一个满足条件的元素

### 感悟与总结（1个）
- 单调栈是解决"下一个更大/更小"类问题的利器，它将暴力解法中的嵌套循环优化为线性时间。今天通过多个实际问题的练习，深刻体会到算法优化的威力，以及不同问题背后统一的思想

### 重点知识点
- **单调递增栈**：栈底到栈顶元素递增，用于寻找"下一个更小/相等元素"
- **单调递减栈**：栈底到栈顶元素递减，用于寻找"下一个更大/相等元素"
- **柱状图最大矩形**：对每个柱子，找到左右第一个比它矮的柱子，计算面积
- **每日温度**：从右向左遍历，利用已计算结果进行跳跃查找（动态规划思想）
- **商品最终价格**：从右向左维护单调递增栈，找到第一个折扣价
- **有效括号**：遇到开括号入栈，遇到闭括号检查栈顶是否匹配
- **时间复杂度分析**：暴力解法O(n²)，单调栈O(n)，动态规划跳跃法O(n)

**LeetCode 84 - 柱状图中最大的矩形（暴力解法）**
```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int res = 0;
        int n = heights.size();
        for(int mid=0;mid<n;++mid){
            int h=heights[mid];
            int left=mid,right=mid;
            while(left-1>=0&&heights[left-1]>=h){
                --left;
            }
            while(right+1<n&&heights[right+1]>=h){
                ++right;
            }
            res=max(res,(right-left+1)*h);
        }
        return res;
    }
};
```

**LeetCode 84 - 柱状图中最大的矩形（单调栈解法 - 两次遍历）**
```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int res = 0;
        int n = heights.size();
        vector<int>left(n),right(n);

        vector<int>stk;
        for(int i=0;i<n;++i){
            while(!stk.empty()&&heights[i]<=heights[stk.back()]){
                stk.pop_back();
            }
            left[i]=(stk.empty()?-1:stk.back());
            stk.push_back(i);
        }
        stk=vector<int>();
        for(int i=n-1;i>=0;--i){
            while(!stk.empty()&&heights[i]<=heights[stk.back()]){
                stk.pop_back();
            }
            right[i]=(stk.empty()?n:stk.back());
            stk.push_back(i);
        }
        for(int i=0;i<n;++i){
            res=max(res,(right[i]-left[i]-1)*heights[i]);
        }
        return res;
    }
};
```

**LeetCode 84 - 柱状图中最大的矩形（单调栈解法 - 一次遍历）**
```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int res = 0;
        int n = heights.size();
        vector<int> left(n), right(n, n);
        vector<int> stk;

        // 优化点1: 预分配内存，避免 vector 动态扩容开销
        stk.reserve(n);

        for (int i = 0; i < n; ++i) {
            // 维护单调递增栈
            while (!stk.empty() && heights[i] <= heights[stk.back()]) {
                right[stk.back()] = i; // 记录右边界
                stk.pop_back();
            }
            left[i] = (stk.empty() ? -1 : stk.back()); // 记录左边界
            stk.push_back(i);
        }

        // 计算最大面积
        for (int i = 0; i < n; ++i) {
            res = max(res, (right[i] - left[i] - 1) * heights[i]);
        }
        return res;
    }
};
```

**LeetCode 739 - 每日温度（单调栈解法）**
```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> stk;
        vector<int> res(n);
        for (int i = n - 1; i >= 0; --i) {
            int t = temperatures[i];
            while (!stk.empty() && temperatures[stk.back()] <= t) {
                stk.pop_back();
            }
            res[i] = stk.empty() ? 0 : stk.back() - i;
            stk.push_back(i);
        }
        return res;
    }
};
```

**LeetCode 739 - 每日温度（动态规划跳跃法）**
```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> res(n);
        for (int i = n - 2; i >= 0; --i) {
            int j = i + 1;
            if (temperatures[i] < temperatures[j]) {
                res[i] = 1;
                continue;
            }
            while (res[j] != 0) {
                if (temperatures[i] < temperatures[j + res[j]]) {
                    res[i] = res[j] + j - i;
                    break;
                } else {
                    j += res[j];
                }
            }
        }
        return res;
    }
};
```

```cpp
class Solution {
public:
    vector<int> finalPrices(vector<int>& prices) {
        int l=prices.size();
        for(int i=0;i<l-1;++i){
            for(int j=i+1;j<l;++j){
                if(prices[j]<=prices[i]){
                    prices[i]-=prices[j];
                    break;
                }
            }
        }
        return prices;
    }
};
```

**LeetCode 1475 - 商品最终价格（单调栈解法）**
```cpp
class Solution {
public:
    vector<int> finalPrices(vector<int>& prices) {
        vector<int> stk;
        vector<int> ans(prices.size());
        stk.reserve(prices.size());
        for (int i = prices.size() - 1; i >= 0; --i) {
            int num = prices[i];
            while (!stk.empty() && stk.back() > num) {
                stk.pop_back();
            }
            ans[i] = stk.empty() ? prices[i] : prices[i] - stk.back();
            stk.emplace_back(prices[i]);
        }
        return ans;
    }
};
```

**LeetCode 20 - 有效括号（栈基础应用）**

```cpp
class Solution {
public:
    bool isValid(const string& s) {
        if (s.size() & 1) {
            return false;
        }
        vector<char> stk;
        //stk.reserve(s.size());
        for (char c : s) {
            if (c == '{' || c == '[' || c == '(') {
                stk.push_back(c);
            } else {
                if (stk.empty()) {
                    return false;
                }
                char top = stk.back();
                stk.pop_back();
                if ((c == '}' && top != '{') ||
                    (c == ']' && top != '[') ||
                    (c == ')' && top != '(')) {
                    return false;
                }
            }
        }
        return stk.empty();
    }
};
```
